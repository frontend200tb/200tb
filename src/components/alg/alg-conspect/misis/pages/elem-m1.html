Ввод и вывод целого числа.

#include <iostream>

using namespace std;

int main() {
  int a;

Ввод трех целых чисел и вывод их произведения.

#include <iostream>

using namespace std;

int main() {
  int a, b, c;
  cin >> a >> b >> c;
  cout << a * b * c;
}
