Задача 7 из 13. Линейное уравнение

Решение 
#include <iostream>

using namespace std;

int main() {
  int a, b;
  cin >> a >> b;

  if (a == 0) {
    if (b != 0) {
      cout << "NO";
    } else {
      cout << "INF";
    }
  } else {
    if (b % a == 0) {
      cout << -b / a;
    } else {
      cout << "NO";
    }
  }

  return 0;
}

Задача 263 Метро с сайта acmp.ru

Решение 
#include <iostream>

using namespace std;

int main() {
  int n, i, j;
  cin >> n >> i >> j;

  int temp = abs(i - j);
  if (temp < n - temp) {
    cout << temp - 1;
  } else {
    cout << n - temp - 1;
  }

  return 0;
}

Задача 854 Кондиционер с сайта acmp.ru

Решение

Это задача на "разбор случаев". В таких задачах нужно проверить каждый вариант.

#include <iostream>

using namespace std;

int main() {
  int t1, t2;
  cin >> t1 >> t2;
  string s;
  cin >> s;

  int res = 0;

  if (s == "freeze") {
    if (t1 >= t2) {
      res = t2;
    } else {
      res = t1;
    }
  }

  if (s == "heat") {
    if (t1 <= t2) {
      res = t2;
    } else {
      res = t1;
    }
  }

  if (s == "auto") {
    res = t2;
  }

  if (s == "fan") {
    res = t1;
  }

  cout << res;

  return 0;
}
