Задача 340 Коробки с сайта acmp.ru

Решение

Нужно расположить стороны коробок в порядке возрастания и сравнить стороны одной коробки со сторонами другой.

#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  int a1, b1, c1, a2, b2, c2; // стороны двух коробок
  cin >> a1 >> b1 >> c1 >> a2 >> b2 >> c2;

  int min1 = min(a1, min(b1, c1));
  int max1 = max(a1, max(b1, c1));
  int mid1 = a1 + b1 + c1 - min1 - max1;

  int min2 = min(a2, min(b2, c2));
  int max2 = max(a2, max(b2, c2));
  int mid2 = a2 + b2 + c2 - min2 - max2;

  if (min1 == min2 and mid1 == mid2 and max1 == max2) {
    cout << "Boxes are equal";
  } else if (min2 >= min1 and mid2 >= mid1 and max2 >= max1) {
    cout << "The first box is smaller than the second one";
  } else if (min1 >= min2 and mid1 >= mid2 and max1 >= max2) {
    cout << "The first box is larger than the second one";
  } else {
    cout << "Boxes are incomparable";
  }
}

Задача 677 Количество участников олимпиады с сайта acmp.ru

Решение

Составим уравнение, где x - число участников

1 / k + 1 / n + 1/ m + d / x = 1

Умножим все на x*k*n*m

x*n*m + x*k*m + x*k*n + k*n*m*d = x*k*n*m k*n*m*d = x*(k*n*m-n*m-k*m-k*n) x = (k*n*m*d)/(k*n*m-n*m-k*m-k*n) 
#include <iostream>

using namespace std;

int main() {
  int k, n, m, d;
  cin >> k >> n >> m >> d;

  int num = k * n * m * d; // числитель
  int den = k * n * m - n * m - k * m - k * n; // знаменатель

  if (den <= 0 or num % den != 0) {
    cout << -1;
  } else {
    int res = num / den;
    if (res % k != 0 or res % n != 0 or res % m != 0) {
      cout << -1;
    } else {
      cout << res;
    }
  }
}
