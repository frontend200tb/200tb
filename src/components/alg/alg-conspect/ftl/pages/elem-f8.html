Задача 11 из 13. Проездной

Решение

Если все билеты разделить на 60 и остаток от деления больше или равен 35, то берем проездные только по 60.

#include <iostream>

using namespace std;

int main() {
  int n;
  cin >> n;
  int cnt1 = 0, cnt10 = 0;, cnt60 = 0;

  if (n % 60 >= 35) {
    cnt60 = n / 60 + 1;
  } else {
    cnt60 = n / 60;
    n %= 60;
    if (n % 10 == 9) {
      cnt10 = n / 10 + 1;
    } else {
      cnt10 = n / 10;
      cnt1 = n % 10;
    }
  }

  cout << cnt1 << cnt10 << cnt60;
  return 0;
}
