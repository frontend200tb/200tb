Теперь пусть числа задает пользователь программы. Наша программа живет в кабинете в котором есть два конвейера. По одному конвейеру числа приходят в нашу программу, а по другому уходят. Переменные это подписанные коробки. Объявить переменную это взять коробку и написать на ней название. Инициализировать переменную это все выбросить из коробки и положить туда какое-то значение. После объявления и инициализации, переменные могут участвовать в арифметических выражениях

При объявлении переменной, нужно задать ее тип. Это как подобрать нужный размер коробочки для значения которое будет в ней храниться. Int имеет размер 32 бита.

06 Действительные числа

Сегодня лекция про вещественные числа. Например, масса земли

5.972*1024 кг

Это экспоненциальная запись числа, где 24 это экспонента, а 5.972 это мантисса. Не все вещественные числа можно хранить в памяти компьютера. Например, число пи имеет бесконечное число знаков и все их сохранить не получится - память просто кончится. Числа хранятся в виде двух целых чисел - мантиссы и экспоненты и еще один бит отводится на знак плюс или минус.

Задача 1

Подсчитаем объем пиццы с радиусом r и высотой h

#include <iostream>

using namespace std;

int main() {
  const double pi = 3.1415;
  double r, h;
  cin >> r >> h;
  cout << pi * r * r * h;

  return 0;
}

По умолчанию программа выводит действительные числа в удобном ей формате. Т.е. это может быть мантиса с экспонентой. Чтобы выводить в обычном формате подключим библиотеку iomanip. Перед выводом укажем fixed и установим точность setprecision

#include <iostream>
#include <iomanip>

using namespace std;

int main() {
  const double pi = 3.1415;
  double r, h;
  cin >> r >> h;
  cout << fixed << setprecision(3);
  cout << pi * r * r * h;

  return 0;
}

Если в арифметическом выражении есть целые и вещественные числа то целые числа преобразуются в вещественные и операции проходят как над вещественными числами.

#include <iostream>

using namespace std;

int main() {
	int a = 1, b = 2;
	double x = 2.5, y = 3.5;
	cout << a / x;

  return 0;
}

Чтобы при делении двух целых чисел получить вещественный результат, то одно из чисел нужно привести к вещественному

#include <iostream>

using namespace std;

int main() {
  int a = 1, b = 2;
  cout << (double) a / b;

  return 0;
}

Задача 4

Считать последовательность из n чисел. Поменять местами начальный элемент и минимальный. Вывести последовательность

#include <iostream>
#include <vector>

using namespace std;

int main() {
  int n;
  cin >> n;
  vector<int> a;
  for (int i = 0; i < n; i++) {
    int temp;
    cin >> temp;
    a.push_back(temp);
  }
  int idx_min = 0;
  for (int i = 1; i < n; i++) {
    if (a[i] < a[idx_min]) {
      idx_min = i;
    }

  a[idx_min] = tmp;

    cout << now << " ";
  }

}

Задача 5

Отсортировать последовательность

#include <iostream>
#include <vector>

using namespace std;

int main() {
  int n;
  cin >> n;
  vector<int> a;
  for (int i = 0; i < n; i++) {
    int temp;
    cin >> temp;
    a.push_back(temp);
  }

  for (int j = 0; j < n - 1; j++) {
    int idx_min = j;
    for (int i = j + 1; i < n; i++) {
      if (a[i] < a[idx_min]) {
        idx_min = i;
      }
    }
    int tmp = a[j];
    a[j] = a[idx_min];
    a[idx_min] = tmp;
  }

  for (auto now : a) {
    cout << now << " ";
  }

  return 0;
}

08 Двумерные массивы

Задача 1

Нарисуем флаг такого вида

1 0 0 0 0
2 1 0 0 0
2 2 1 0 0
2 2 2 1 0
2 2 2 2 1

#include <iostream>

using namespace std;

int main() {
  int n;
  cin >> n;
  int a[100][100];

  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      if (i == j) {
        a[i][j] = 1;
      } else if (i < j) {
        a[i][j] = 0;
      } else {
        a[i][j] = 2;
      }
    }
  }

  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      cout << a[i][j] << " ";
    }
    cout << endl;
  }

  return 0;
}

Задача 2

Дана матрица с расставлеными минами, 0 - нет мины, 1 - есть мина. Сгенерировать поле для игры в сапера. Поставить 0 если в соседних клетках нет мин. Поставить цифру от 1 до 8 равную числу мин в соседних 8 клетках. Поставить * если в этой клетке мина. В первой строке на вход приходят два числа n и m (не более 100) это строки и столбцы, размер поля. Далее приходит матрица n на m

Считаем входные данные и положим их в массив с рамочкой, т.е. вокруг массива будут пустые поля с нулями.

#include <iostream>

using namespace std;

int main() {
  int n, m;
  cin >> n >> m;

  // заполнение нулями
  int a[102][102];
  for (int i = 0; i <= n + 1; i++) {
    for (int j = 0; j <= m + 1; j++) {
      a[i][j] = 0;
    }
  }

  // ввод данных
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      cin >> a[i][j];
    }
  }

  // заполнение поля
  int ans[102][102];
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      // сдвиги для координат соседей
      int dx[8] = { 1, 1, 1, 0, 0, -1, -1, -1 };
      int dy[8] = { -1, 0, 1, -1, 1, 0, 1, -1 };
      int tmp = 0;
      // перебор соседей
      for (int k = 0; k < 8; k++) {
        tmp += a[i + dy[k]][j + dx[k]];
      }
      ans[i][j] = tmp;
    }
  }

  // вывод
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      if (a[i][j] == 1) {
        cout << "*";
      } else {
        cout << ans[i][j];
      }
    }
    cout << endl;
  }

  return 0;
}
