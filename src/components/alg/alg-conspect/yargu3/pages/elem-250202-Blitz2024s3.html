    // решение

    if (n % 3 == 0) {
      h2 = n / 3;
      h1 = n / 3 + 1;
      h3 = n / 3 - 1;
    }

    if (n % 3 == 1) {
      h2 = n / 3;
      h1 = n / 3 + 2;

    }

    }

    // вывод результата
    cout << h2 << ' ' << h1 << ' ' << h3 << '\n';
  }

}

Задача B. Декременты массива

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
      cin >> A[i];
    }
    vector<int> B(n);
    for (int i = 0; i < n; i++) {
      cin >> B[i];
    }

    // решение

    // найдем максимальное число операций
    int step = 0;
    int max_step = step;
    for (int i = 0; i < n; i++) {
      step = A[i] - B[i];
      max_step = max(max_step, step);
    }

    bool res = true;

    // число операций должно быть максимальное если B[i] не равно 0
    for (int i = 0; res && i < n; i++) {
      step = A[i] - B[i];
      if (B[i] == 0) {
        continue;
      }
      if (step < 0 || step < max_step) {
        res = false;
      }
    }

    // вывод результата
    if (res) {
      cout << "YES\n";
    } else {
      cout << "NO\n";
    }

  }

}
