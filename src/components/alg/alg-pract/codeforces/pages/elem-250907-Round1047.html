Задача B. Веселая перестановка

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

  // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // размер перестановки
    cin >> n;
    vector<int> p(n + 1);
    for (int i = 1; i <= n; i++) {
      cin >> p[i];
    }

    // решение
    vector<int> q(n + 1);

    // обратная перестановка
    for (int i = 1; i <= n; i++) {
      q[i] = n - p[i] + 1;
    }

    // вывод результата
    for (int i = 1; i <= n; i++) {
      cout << q[i] << ' ';
    }
    cout << '\n';
  }
}

Задача C. Максимальная четная сумма

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

typedef long long ll;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

  // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    ll a, b; // два целых числа
    cin >> a >> b;

    // решение
    ll res = -1;

    if (a * b % 2 == 1) {
      res = a * b + 1;
    } else if (b % 2 == 0 && a * b % 4 == 0) {
      res = a * b / 2 + 2;
    }

    // вывод результата
    cout << res << '\n';
  }
}
