Задача A. Превращение: из A в B

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <stack>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // ввод данных
  int a; // число которое есть
  int b; // число которое надо получить
  cin >> a >> b;

  // решение
  stack<int> Res; // последовательность превращений

  // пойдем в обратную сторону
  // будем получать из числа b число a
  Res.push(b);
  int cnt = 1; // количество чисел в превращении

  // пока можно уменьшать число b
  while (b > a) {
    if (b % 10 == 1) {
      b /= 10;
      Res.push(b);
      cnt++;
    } else if (b % 2 == 0) {
      b /= 2;
      Res.push(b);
      cnt++;
    } else {
      // это невозможно
      cout << "NO";
      return 0;
    }
  }

  // вывод результата
  if (b == a) {
    cout << "YES" << '\n' << cnt << '\n';
    while (!Res.empty()) {
      cout << Res.top() << ' ';
      Res.pop();
    }
  } else {
    cout << "NO";
    return 0;
  }

}
