Задача A. Редстоун?

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // число шестеренок
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
      cin >> A[i];
    }

    // решение
    bool noDuplicate = true;
    set<int> st;
    for (int i = 0; noDuplicate && i < n; i++) {
      if (st.find(A[i]) != st.end()) {
        noDuplicate = false;
      } else {
        st.insert(A[i]);
      }
    }


    // вывод результата
    if (noDuplicate) {
      cout << "NO\n";
    } else {
      cout << "YES\n";
    }

  }

}

Задача B. Жители

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // число жителей
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
      cin >> A[i]; // недовольство каждого жителя
    }

    // решение
    ll res = 0;
    sort(A.begin(), A.end());

    // раздаем изумруды от большего к меньшему попарно
    // потом дружим тех у кого их 0
    for (int i = n - 1; i > 0; i -= 2) {
      int inc = A[i];
      int dec = A[i - 1];
      A[i] = A[i] - dec;
      A[i - 1] = 0;
      res = res + inc;
    }

    // проверяем последнего жителя
    // если их было нечетное количество
    if (A[0] > 0) {
      res += A[0];
    }

    // вывод результата
    cout << res << '\n';
  }

}

Задача C. Нижний мир

Ограничение по времени 3с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> d(n+1);
        for (int i = 1; i <= n; i++) {
            cout << "? " << i << " " << n;
            for (int j = 1; j <= n; j++) {
                cout << " " << j;
            }
            cout << endl;
            cin >> d[i];
            if (d[i] == -1) {
                return 0;
            }
        }

        vector<vector<int> > groups(n+1);
        int max_d = 0;
        for (int i = 1; i <= n; i++) {
            if (d[i] > max_d) {
                max_d = d[i];
            }
            groups[d[i]].push_back(i);
        }

        vector<int> path;
        int current = groups[max_d][0];
        path.push_back(current);

        for (int l = max_d - 1; l >= 1; l--) {
            for (int u : groups[l]) {
                cout << "? " << current << " 2 " << current << " " << u << endl;
                int res;
                cin >> res;
                if (res == -1) {
                    return 0;
                }
                if (res == 2) {
                    current = u;
                    path.push_back(u);
                    break;
                }
            }
        }

        cout << "! " << path.size();
        for (int node : path) {
            cout << " " << node;
        }
        cout << endl;
    }
    return 0;
}

Задача D. Наездник на курице

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	// число тестов
	int t;
	cin >> t;

	while (t--)	{
		// ввод данных
		int n; // число мобов
		cin >> n;
		vector<int> A(n);
		for (int i = 0; i < n; i++) {
			cin >> A[i]; // здоровье каждого моба
		}

		// решение
		vector<long long> DP(n);
		long long sum = 0;

		// динамическое программирование
		for (int i = n - 1; i >= 0; i--) {
			sum += A[i];
			if (i > 0) {
				DP[i - 1] = max(DP[i - 1], DP[i] + 1);
			}
			if (i > 1) {
				DP[i - 2] = max(DP[i - 2], DP[i] + min(i, A[i]));
			}
		}

		// вывод результата
		cout << sum - DP[0] << "\n";
	}
}
