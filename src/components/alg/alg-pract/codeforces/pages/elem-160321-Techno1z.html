Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  int n, k;
  cin >> n >> k;

  // решение
  int res = 6 - n - k;

  // вывод результата
  cout << res;
}
