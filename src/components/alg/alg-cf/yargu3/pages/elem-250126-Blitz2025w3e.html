Задача A. Массив

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ввод данных
  int n; // длина массива
  cin >> n;
  vector<int> A(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i];
  }

  // решение
  vector<int> Neg; // числа с отрицательным произведением
  vector<int> Pos; // числа с положительным произведением
  vector<int> Zer; // числа с нулевым произведением

  // возьмем одно отрицательное число для массива Neg
  // возьмем одно положительное или
  // два отрицательных числа для массива Pos
  // среди остальных чисел должен быть 0, поэтому
  // все остальные числа возьмем в массив Zer

  for (int i = 0; i < n; i++) {

    if (Neg.size() == 0 && A[i] < 0) {
      Neg.push_back(A[i]);
      continue;
    }

    if (Pos.size() == 0 && A[i] != 0) {
      Pos.push_back(A[i]);
      continue;
    }

    if (Pos.size() == 1 && Pos[0] < 0 && A[i] > 0) {
      Zer.push_back(Pos[0]);
      Pos.pop_back();
      Pos.push_back(A[i]);
      continue;
    }

    if (Pos.size() == 1 && Pos[0] < 0 && A[i] < 0) {
      Pos.push_back(A[i]);
      continue;
    }

    Zer.push_back(A[i]);
  }

  // вывод результата
  cout << Neg.size() << ' ' << Neg[0] << '\n';
  cout << Pos.size() << ' ';
  for (int i = 0; i < Pos.size(); i++) {
    cout << Pos[i] << ' ';
  }
  cout << '\n';
  cout << Zer.size() << ' ';
  for (int i = 0; i < Zer.size(); i++) {
    cout << Zer[i] << ' ';
  }
  cout << '\n';
}
