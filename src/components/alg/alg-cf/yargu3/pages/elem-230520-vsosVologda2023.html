Задача 1. Ромбический порядок

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // ввод данных
  int x; // число шагов по горизонтали
  int y; // число шагов по вертикали
  cin >> x >> y;

  // решение
  long long res = 1;
  long long romb = abs(x) + abs(y); // номер ромба
  long long romb_begin = 1; // с какого числа начинается ромб
  long long romb_size = 1; // сколько чисел в ромбе

  // находим с какого числа начинается ромб и сколько в нем чисел
  for (long long i = 1; i <= romb; i++) {
    romb_begin = romb_begin + romb_size;
    romb_size = (i == 1 ? 4 : romb_size + 4);
  }

  // находим в какой четверти число
  if (x >= 0 && y >= 0) {
    // первая четверть
    res = romb_begin + x;
  } else if (x < 0 && y >= 0) {
    // вторая четверть
    res = romb_begin + romb_size + x;
  } else if (x < 0 && y < 0) {
    // третья четверть
    res = romb_begin + romb_size / 2 + abs(x);
  } else if (x >= 0 && y < 0) {
    // четвертая четверть
    res = romb_begin + romb_size / 2 - x;
  }

  // вывод результата
  cout << res;
}
