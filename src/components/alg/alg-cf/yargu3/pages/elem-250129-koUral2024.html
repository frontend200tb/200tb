Задача A. Билли Бонс и монеты

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

int main() {
  // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // ввод данных
  long long n; // число
  cin >> n;

  // решение
  long long m = 0;
  long long digit = 0;
  long long pos = 1;

  // находим младшие разряды
  while (n > 9) {
    digit = n % 10;
    m += (digit == 0 ? 1 : 0) * pos;
    pos *= 10;
    n /= 10;
  }

  // находим старший разряд
  digit = n + 1;
  m += digit * pos;

  // вывод результата
  cout << m;
}
