Задача 1. Змейка

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // ввод данных
  long long n; // сторона квадрата
  cin >> n;

  // решение
  long long res = 0;
  for (long long i = 1; i <= n; i++) {
    res += 2 * i + 1;
  }

  // вывод результата
  cout << res;
}

Задача B. Две сестры

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

typedef unsigned long long ull;

int main() {
  // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  // ввод данных
  long long a; // 1 таблетка в a дней
  long long b; // b таблеток каждый день
  long long n; // число таблеток
  cin >> a >> b >> n;

  // решение
  long long left = 0;
  long long right = n / b;

  while (right - left > 1) {
    long long mid = left + (right - left) / 2;
    long long atabl = (mid + a - 1) / a;
    long long btabl = b * mid;
    long long total = atabl + btabl;

    if (total <= n) {
      left = mid;
    } else {
      right = mid;
    }
  }

  // вывод результата
  cout << left;
}
