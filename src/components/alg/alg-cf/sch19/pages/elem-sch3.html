#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ввод данных
  int n; // число комнат
  cin >> n;
  vector<int> A(n);
  vector<int> B(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i]; // число студентов в комнате
    cin >> B[i]; // число мест в комнате
  }

  // решение
  int res = 0; // число комнат с 2 свободными местами
  for (int i = 0; i < n; i++) {
    if (B[i] - A[i] > 1) {
      res++;
    }
  }

  // вывод результата
  cout << res;
}
