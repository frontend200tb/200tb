#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ввод данных
  int n; // число комнат
  cin >> n;
  vector<int> A(n);
  vector<int> B(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i]; // число студентов в комнате
    cin >> B[i]; // число мест в комнате
  }

  // решение
  int res = 0; // число комнат с 2 свободными местами
  for (int i = 0; i < n; i++) {
    if (B[i] - A[i] > 1) {
      res++;
    }
  }

  // вывод результата
  cout << res;
}

Задача D. Халк

Решение
Задача 705A
Codeforces 705 Round 366 (Div. 2) 2010.12.20 
#include <iostream>
#include <string>

using namespace std;

int main() {
  // ввод данных
  int n; // число уровней
  cin >> n;

  // решение
  string res = "";
  string start = "I hate ";
  string hate = "that I hate ";
  string love = "that I love ";
  string end = "it";

  // собираем строку
  res.append(start);

  for (int i = 1; i < n; i++) {
    if (i % 2 == 0) {
      res.append(hate);
    } else {
      res.append(love);
    }
  }

  res.append(end);

  // вывод результата
  cout << res << '\n';
}
