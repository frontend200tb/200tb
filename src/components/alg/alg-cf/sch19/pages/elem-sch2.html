#include <iostream>
#include <set>

using namespace std;

int main() {
  // ввод данных
  int s1, s2, s3, s4; // цвета подков
  cin >> s1 >> s2 >> s3 >> s4;

  // решение
  set<int> F;
  F.insert(s1);
  F.insert(s2);
  F.insert(s3);
  F.insert(s4);

  // вывод результата

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <set>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива
    cin >> n;
    int in;

    // соберем набор неповторяющихся элементов
    set<int> st;
    for (int i = 0; i < n; i++) {
      cin >> in;
      st.insert(in);
    }

    // вывод результата
    cout << st.size() << '\n';
  }

}
