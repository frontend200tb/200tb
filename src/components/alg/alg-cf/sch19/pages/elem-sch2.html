  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива
    cin >> n;
    int in;

    // соберем набор неповторяющихся элементов

  }

}

Задача D. Чет и нечет

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение
Задача 318A

#include <iostream>

using namespace std;

int main() {
  // ввод данных
  long long n, k;
  cin >> n >> k;

  // решение
  long long res = 1;
  if (n % 2 == 0) {
    if (k <= n / 2) {
      res = k * 2 - 1;
    } else {
      res = (k - n / 2) * 2;
    }
  } else {
    if (k <= n / 2 + 1) {
      res = k * 2 - 1;
    } else {
      res = (k - n / 2 - 1) * 2;
    }
  }

  // вывод результата
  cout << res;
}

Codeforces 34 Round 34 Beta (Div. 2) 2010.10.11 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  // ввод данных
  int n; // число тв
  int m; // число сколько можно взять тв
  cin >> n >> m;
  vector<int> A(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i];
  }

  // решение
  sort(A.begin(), A.end());
  int res = 0;
  int i = 0;
  while (i <= m and A[i] < 0) {
    res -= A[i];
    i++;
  }

  // вывод результата
  cout << res;
}
