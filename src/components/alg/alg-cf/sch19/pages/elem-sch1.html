Задача A. Трамвай

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

  // ввод данных
  int n; // число остановок
  cin >> n;
  vector<int> A(n + 1);
  vector<int> B(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i] >> B[i];
  }

  // решение
  int cap = 0;
  int capm = 0;

  for (int i = 1; i <= n; i++) {
    cap -= A[i];
    cap += B[i];
    if (cap > capm) {
      capm = cap;
    }
  }

  // вывод результата
  cout << capm;
}

Задача C. Суперагент

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение

Создаем массив символов на 10 элементов, нулевой элемент инициализируем чем угодно. Введем 9 символов в элементы массива с 1 до 9. Для симметрии 5 элемент может быть любым. Проверим что равны элементы массива A[1] = A[9], A[2] = A[8], A[3] = A[7], A[4] = A[6].

#include <iostream>

using namespace std;

int main() {
  // ввод данных
  char A[10];
  A[0] = '0';
  for (int i = 1; i < 10; i++) {
    cin >> A[i];
  }

  // решение
  bool flag = true;
  for (int i = 1; i < 5; i++) {
    if (A[i] != A[10 - i]) {
      flag = false;
    }
  }

  // вывод результата
  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}

Задача D. Правильное решение?

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение

Поскольку ответ Валеры может содержать лидирующие нули, то примем два числа как строки. Сразу проверим, если у Валеры больше одного символа и первый символ 0, то это неправильный ответ. Соберем частотный массив для цифр от 0 до 9. Соберем свой правильный ответ. Первую цифру поставим наименьшую, но не 0. Следующие цифры должны идти от 0 до 9, чтобы получилось наименьшее число.

#include <iostream>
#include <string>

using namespace std;

int main() {
  // ввод данных
  string sn; // целое число
  string sm; // мин число
  cin >> sn >> sm;

  // решение
  bool flag = true;

  if (sm.size() > 1 && sm[0] == '0') {
    flag = false;
  }
  long long n = stoll(sn);
  long long m = stoll(sm);
  // соберем частотный массив цифр числа
  int F[10]{ 0 };
  long long tmp = n;
  while (tmp) {
    int dig = tmp % 10;
    F[dig]++;
    tmp /= 10;
  }

  // соберем свое мин число k
  // оно может и не влезть в int
  // первая цифра мин, но не 0
  long long k = 0;
  int next = 1;
  while (k == 0 && next < 10) {
    if (F[next] > 0) {
      k = next;
      F[next]--;
    }
    next++;
  }

  // остальные цифры мин начиная с нуля
  for (int i = 0; i < 10; i++) {
    while (F[i]) {
      k = k * 10 + i;
      F[i]--;
    }
  }

  // сравним результаты
  if (k != m) {
    flag = false;
  }

  // вывод результата
  if (flag) {
    cout << "OK";
  } else {
    cout << "WRONG_ANSWER";
  }
}
