Задача A. Трамвай

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  // ускорение ввода-вывода
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);

  // ввод данных
  int n; // число остановок
  cin >> n;
  vector<int> A(n + 1);
  vector<int> B(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i] >> B[i];
  }

  // решение
  int cap = 0;
  int capm = 0;

  for (int i = 1; i <= n; i++) {
    cap -= A[i];
    cap += B[i];
    if (cap > capm) {
      capm = cap;
    }
  }

  // вывод результата
  cout << capm;
}
