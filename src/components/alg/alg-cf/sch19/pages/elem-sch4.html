Решение
Задача 339A
Codeforces 339 Round 197 Beta (Div. 2) 2013.08.26 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  string s; // строка вида 1+2+3+2+1 из цифр 1,2,3
  cin >> s;

  // решение
  int n = s.size();
  int cnt1 = 0; // число единиц
  int cnt2 = 0; // число двоек
  int cnt3 = 0; // число троек

  for (int i = 0; i < n; i += 2) {
    if (s[i] == '1') {
      cnt1++;
    }
    if (s[i] == '2') {
      cnt2++;
    }
    if (s[i] == '3') {
      cnt3++;
    }
  }

  char res[100];
  int pos = 0;
  for (int i = 0; i < cnt1; i++) {
    res[pos++] = '1';
    res[pos++] = '+';
  }
  for (int i = 0; i < cnt2; i++) {
    res[pos++] = '2';
    res[pos++] = '+';
  }
  for (int i = 0; i < cnt3; i++) {
    res[pos++] = '3';
    res[pos++] = '+';
  }
  if (pos > 0 && res[pos - 1] == '+') {
    pos--;
  }

  // вывод результата
  for (int i = 0; i < pos; i++) {
    cout << res[i];
  }
}
