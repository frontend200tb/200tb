Задача A. Во сне

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int a, b; // счет в первом тайме
    int c, d; // счет во втором тайме
    cin >> a >> b >> c >> d;

    // решение
    bool flag = true;

    if (a > 2 * b + 2) {
      flag = false;
    }

    if (b > 2* a + 2) {
      flag = false;
    }

    if (c - a > 2 * (d - b) + 2) {
      flag = false;
    }

    if (d - b > 2 * (c - a) + 2) {
      flag = false;
    }

    // вывод результата
    if (flag) {
      cout << "YES\n";
    } else {
      cout << "NO\n";
    }
  }
}

Задача B. Как битсет

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина бинарной строки
    int k; // целое число
    cin >> n >> k;
    string s; // бинарная строка
    cin >> s;

    // решение
    bool flag = true;

    // посчитаем сколько в строке единиц
    int cnt1 = 0; // число единиц в строке
    for (int i = 0; i < n; i++) {
      if (s[i] == '1') {
        cnt1++;
      }
    }

    // для всех чисел от 1 до n соберем перестановку
    // такую что наименьшие числа от 1 до cnt1 будут в s[i] = '1'
    int p1 = 1; // первое число которое пойдет в s[i] = '1'
    int p0 = cnt1 + 1; // первое число которое пойдет в s[i] = '0'

    // соберем оптимальную перестановку
    vector<int> P(n);
    for (int i = 0; i < n; i++) {
      if (s[i] == '1') {
        P[i] = p1;
        p1++;
      } else {
        P[i] = p0;
        p0++;
      }
    }

    // проверим перестановку на максимум в s[i] = '1'
    // используем метод скользящего окна
    // число подряд идущих единиц должно быть меньше k
    int cnt1k = 0;
    for (int i = 0; i < k; i++) {
      if (s[i] == '1') {
        cnt1k++;
      }
    }
    if (cnt1k == k) {
      flag = false;
    }

    for (int i = k; flag && i < n; i++) {
      if (s[i - k] == '1') {
        cnt1k--;
      }
      if (s[i] == '1') {
        cnt1k++;
      }
      if (cnt1k == k) {
        flag = false;
      }
    }

    // вывод результата
    if (flag) {
      cout << "YES\n";
      for (int i = 0; i < n; i++) {
        cout << P[i] << ' ';
      }
      cout << '\n';
    } else {
      cout << "NO\n";
    }
  }
}

Задача C. Вопреки различиям

Ограничение по времени 2с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) {
      cin >> a[i];
    }

    // решение
    vector<vector<int> > pos(n + 1);
    vector<int> dp(n + 2, 0);
    vector<int> s(n + 1);

    for (int i = 0; i < n; i++) {
      s[i] = pos[a[i]].size();
      pos[a[i]].push_back(i);
    }

    for (int i = n - 1; i >= 0; i--) {
      if (s[i] + a[i] - 1 < pos[a[i]].size()) {
        dp[i] = max(dp[pos[a[i]][s[i] + a[i] - 1] + 1] + a[i], dp[i + 1]);
      } else {
        dp[i] = dp[i + 1];
      }
    }

    int res = dp[0];

    // вывод результата
    cout << res << '\n';
  }
}

    for (int i = 0; i < n; i++) {
      cin >> a[i].first >> a[i].second;
    }

    // решение
    ll res;
    ll m = 1e9;
    cout << "? D " << m << endl;
    cin >> res;
    cout << "? D " << m << endl;
    cin >> res;
    cout << "? L " << m << endl;
    cin >> res;
    cout << "? L " << m << endl;
    cin >> res;
    ll nres;
    cout << "? R " << m << endl;
    cin >> nres;
    cout << "? R " << m << endl;
    cin >> nres;
    cout << "? R " << m << endl;
    cin >> nres;
    cout << "? R " << m << endl;
    cin >> nres;
    ll mn = 3e9;
    ll x1, y1;
    for (int i = 0;i < n;i++) {
      if (a[i].first + a[i].second < mn) {
        mn = min(mn, a[i].first + a[i].second);
        x1 = a[i].first;
        y1 = a[i].second;
      }
    }
    ll h1 = x1 + y1 - res;
    ll nmn = 3e9;
    ll x2, y2;
    for (int i = 0;i < n;i++) {
      if (a[i].second - a[i].first < nmn) {
        nmn = min(nmn, a[i].second - a[i].first);
        x2 = a[i].first;
        y2 = a[i].second;
      }
    }
    ll h2 = nres + x2 - y2 - 4 * m;
    ll ansx = (h1 + h2) / 2;
    ll ansy = (h1 - h2) / 2;
    ansx += 2 * m;
    ansy += 2 * m;
    cout << "! " << ansx << " " << ansy << endl;
  }
}
