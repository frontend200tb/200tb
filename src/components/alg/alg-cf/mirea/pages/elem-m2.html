Задача A. Новогодние вывески

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <cctype>

using namespace std;

int main() {
  // ввод данных
  string a, b; // две строки
  cin >> a >> b;

  // решение
  int res = 0;
  int len = a.size();
  bool flag = true;

  for (int i = 0; flag && i < len; i++) {
    if (tolower(a[i]) > tolower(b[i])) {
      flag = false;
      res = 1;
    }
    if (tolower(b[i]) > tolower(a[i])) {
      flag = false;
      res = -1;
    }
  }

  // вывод результата
  cout << res;
}
