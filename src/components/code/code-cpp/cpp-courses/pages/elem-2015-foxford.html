1 Введение в язык С

Курс посвящен языкам программирования C и C++. Мы начнем с самого начала.

Почему нужно изучать языки C и C++ и чем они лучше других языков?

  1. Скорость. Языки C и C++ это самые быстрые высокоуровневые языки программирования.
  2. Лаконичность. В них нет ничего лишнего, нет избыточных конструкций и этим они достаточно просты. Каждое слово языка и каждый символ существенно влияет на работу программы.
  3. Популярность. Количество кода на языках C и C++ превосходит написанное на любом другом языке.
  4. Обучение. Изначально язык создавался для программистов, для написания ядра операционной системы. Позднее стал использоваться для обучения программированию.
  5. Спортивное программирование.

Язык C разработали в 1969-1973 годах сотрудники Bell Labs Кен Томпсон и Денис Ритчи. Он создавался для разработки операционной системы Unix. Это процедурный язык программирования. Это значит что в программе есть только переменные или функции и нет сложных конструкций.

Язык C++ разработал в 1983 году сотрудник Bell Labs Бьерн Страуструп.

Языки программирования делятся на:

  1. компилируемые (C, C++, Pascal)
  2. интерпретируемые (Python)

Компилятор берет файл с текстом программы на языке C и выдает файл с набором инструкций для процессора.

Интерпретатор построчно проходит по тексту программы и выполняет ее.

Транспонирование матрицы

Транспонирование матрицы это перестановка элементов двумерного массива относительно главной диагонали. При этом матрица размером n x m становится матрицей размером m x n.

#include 

int main() {
  int n, m, i, j;
  scanf("%d %d", &n, &m);

  int a[n][m];
  for (i = 0; i < n; i++) {
    for (j = 0; j < m; j++) {
      scanf("%d", &a[i][j]);
    }
  }

  int b[m][n];
  for (i = 0; i < n; i++) {
    for (j = 0; j < m; j++) {
      b[j][j] = a[i][j];
    }
  }

  for (i = 0; i < m; i++) {
    for (j = 0; j < n; j++) {
      printf("%d", b[i][j]);
    }
    printf("\n");
  }

  return 0;
}

Статический массив остается неизменным во время выполнения программы. Если нам нужно, чтобы число элементов в массиве могло изменяться, то можно написать свой динамический массив. Пусть у нас есть массив A на 10 элементов типа int. Нам нужно массив A увеличить до 20 элементов. Создадим второй массив B на 20 элементов типа int, переместим в него элементы из массива A.

#include 
#include 

int main() {
  // объявляем указатели
  int *a, *b;
  // выделяем память
  a = malloc(sizeof(int) * 10);
  b = malloc(sizeof(int) * 10);

  for (int i = 0; i < 10; i++) {
    // копируем элементы
    b[i] = a[i];
  }

  // освобождаем память
  free(a);

  // переносим указатель на новую область памяти
  a = b;

  return 0;
}

При этом массив a не перезаписывается, просто указатель a теперь указывает на другое место. Память выделенную с помощью функций malloc(), calloc(), realloc() всегда необходимо освобождать с помощью функции free().

Решиим задачу. В первой строке дано число символов, которые мы хотим считать. Во второй строке даны символы, которые нужно посчитать. В третьей строке дана строка с символами.

#include 

typedef struct char_counter {
  char character;
  int count;
} char_counter;

int main() {
  int n;
  scanf("%d", &n);
  char c;
  scanf("%c", &c);

  char_counter res[n];
  for (int i = 0; i < n; i++) {
    scanf("%c", &(res[i].character));
  }

  scanf("%c", &c);
  while (c == 10 || c == 13) {
    scanf("%c", &c);
  }

  while (c != 10 && c != 13) {
    for (int i = 0; i < n; i++) {
      if (res[i].character == c) {
        res[i].count++;
      }
    }
    scanf("%c", &c);
  }

  for (int i = 0; i < n; i++) {
    printf("%c: &d occurences\n", res[i].character, res[i].count);
  }

  return 0;
}

После первого ввода числа n scanf("%d", &n), во входном потоке остается перевод строки, который нужно пропустить scanf("%c", &c). Потом в цикле for считываются символы строки. И опять нужно пропустить символ перевода строки scanf("%c", &c).

Для того чтобы программа понимала символы перевода строки и в windows и в linux, нужна строка while (c != 10 && c != 13)

Напишем функцию abs(), которая принимает целое число и возвращает его абсолютное значение по модулю.

  if (a < 0) {
    a = -a;

  return a;
}

int main() {
  printf("%d\n", abs(-5));

  int x = -10;
  int y = abs(x);
  printf("x = %d y = %d", x, y);

  return 0;
}

Напишем функцию gcd(), которая принимает два целых числа и возвращает их наибольший общий делитель.

#include 

int gcd(int x, int y) {
  while (x != 0 && y != 0) {
    if (x > y) {
      x %= y;
    } else {
      y %= x;
    }
  }
  return x + y;
}

int main() {
  printf("%d", gcd(10, 20));

  return 0;
}

В функции abs() и gcd() мы передавали параметры по значению. При этом внутри функции создавались копии этих параметров, которые могли изменяться. После выполнения функции копии параметров уничтожались, а исходные переменные, которые передавались при вызове функции, оставались неизменными.

Пусть мы хотим создавать функции, которые могут менять значения переменных, для этого мы должны передавать параметры по ссылке или через указатель.

Напишем функцию swap(), которая принимает два указателя на переменные типа int и меняет значение переменных местами.

#include 

void swap(int *x, int *y) {
  int tmp = *x;
  *x = *y;
  *y = tmp;
}

int main() {
  int a, b;
  a = 3;
  b = 5;
  printf("a = %d b = %d\n", a, b);

  swap(&a, &b);
  printf("swap\n");
  printf("a = %d b = %d", a, b);

  return 0;
}

Функция swap() создает две переменные, куда копирует значение входных параметров - адреса переменных x и y. Настоящие адреса, которые мы передали в функцию, сама функция никак изенить не может. Но функция меняет значения, которые записаны по этим адресам.

Напишем функцию array_swap(), которая принимает два массива целых чисел. Если длины массива одинаковые, функция меняет элементы массивов местами и возвращает 0. Если длины массивов разные, то функция возвращает 1.

int array_swap(int n, int m, int *a, int *b) {
  if (n != m) {
    return 1;
  }
  for (int i = 0; i < n; i++) {
    swap(&a[i], &b[i];)
  }
  return 0;
}

Создадим структуру данных для хранения стека. И создадим глобальную переменную current для хранения текущей вершины стека.

struct stack_node {
  int value;
  struct stack_node *next;
};

stack_node *current = 0;

Напишем функции push() И pop() для добавления элемента на вершину стека и удаления элемента с вершины стека.

void push(int x) {
  stack_node *new_n = malloc(sizeof(stack_node));
  new_n->value = x;
  new_n->next = current;
  current = new_n;
}

int pop() {
  int res = current->value;
  stack_node *old = current;
  current = current->next;
  free(old);
  return res;
}

Эти функции push() и pop() работают только с одним стеком, указатель на голову которого лежит в глобальной переменной current. Напишим функции push() и pop(), которые принимают в качестве параметра голову стека и таким образом, могут работать с разными стеками.

Функция push() принимает голову стека и новую вершину. Новую вершину нужно добавить в стек и сделать ее новой головой стека. Функция возвращает новую голову стека.

Функция pop() принимает голову стека и указатель на переменную, куда нужно положить результат. Значение головы стека нужно записать в переменную результата. Голову стека нужно сдвинуть на следующий элемент стека. Функция через параметр возвращает результат. То есть функция принимает указатель на переменную и в него же кладет результат.

stack_node *push(int x, stack_node *head) {
  stack_node *new_n = malloc(sizeof(stack_node));
  new_n->value = x;
  new_n->next = head;
  return new_n;
}

stack_node *pop(stack_node *head, int *res) {
  *res = head->value;
  stack_node *old = head;
  head = head->next;
  free(old);
  return head;
}

Стабильная сортировка сохраняет порядок одинаковых элементов. Нестабильная сортировка может изменять порядок одинаковых элементов.

Напишем функцию merge(), которая соединяет два отсортированных массива в один отсортированный массив. Сделаем это с помощью двух указателей.

int * merge(int n, int *a, int m, int *b) {
  int *res = malloc(n + m);
  int p1 = 0;
  int p2 = 0;
  while (p1 < n || p2 < m) {
    if (p2 == m || a[i] < b[i]) {
      res[i+j] = a[i];
      i++;
    } else {
      res[i+j] = b[j];
      j++;
    }
  }

  free(a);
  free(b);
  return res;
}

Сортировка слиянием

Сортировка слиянием использует алгоритм соединения двух отсортированных массивов в один отсортированный массив. Напишем рекурсивную функцию merge_sort(), которая делит массив на две части для сортировки каждой части отдельно и слияния двух отсортированных частей в один отсортированный массив.

int *merge_sort(int n, int *a) {
  // если в массиве 0 или 1 элемент, то
  // он уже отсортирован и надо его вернуть
  if (n < 2) {
    return a;
  }

  // левая половина массива
  int *left = malloc((n/2)*sizeof(int));
  for (int i = 0; i < n/2; i++) {
    left[i] = a[i];
  }
  // правая половина массива
  int *right = malloc((n-n/2)*sizeof(int));
  for (int i = 0; i < n-n/2; i++) {
    left[i] = a[i+n/2];
  }
  free(a);
  return merge(n/2, merge_sort(n/2, left), n-n/2, merge_sort(n-n/2));
}

Быстрая сортировка

Алгоритм быстрой сортировки выбирает опорный элемент. Желательно чтобы он оказался из середины уже отсортированного массива. Левый указатель, начиная с первого элемента, проходит по всем элементам слева от опорного пока не дойдет до элемента, который больше чем опорный элемент. Правый указатель, начиная с последнего элемента, проходит по всем элементам справа от опорного, пока не дойдет до элемента, который меньше чем опорный элемент. Тогда эти два элемента меняются местами и указатели двигаются дальше. Два указателя двигаются навстречу друг другу. Когда два указателя сойдутся на одном элементе, то получится, что все элементы левее меньше опорного, а все элементы правее больше опорного.

После этого рекурсивно запускаем быструю сортировку для массива, который левее указателей и для массива, который правее указателей.

void qsort(int *a, int l, int r) {
  int p1 = l;
  int p2 = r;
  int mid = a[(l+r)/2];

  while (p1 < p2) {
    while (a[p1] < mid) {
      p1++;
    }
    while (a[p2] > mid) {
      p2--;
    }
    if (p1 <= p2) {
      swap(&a[p1], &a[p2]);
      p1++;
      p2--;
    }
  }

  if (l < p2) {
    qsort(a, l, p2);
  }
  if (p2 < r) {
    qsort(a, p1, l);
  }
}

8 Работа с памятью в С

При индексации массива используется арифметика указателей. К адресу массива прибавляется его индекс. Если при индексации поменять местами имя массива и индекс, то результат останется тем же.

Например, пусть у нас есть массив arr[] и мы хотим обратиться к элементу с индексом 2. Для получения элемента, компилятор произведет разименование указателя, указывающего на адрес этого элемента.

arr[2] = *(arr+2)

Если записать 2[arr], то получится то же самое

2[arr] = *(2+arr) = *(arr+2)

Создадим указатель arr и запишем в него адрес памяти, которую выделит функция malloc для 10 элементов типа int. Теперь с этим указателем arr мы можем работать как с массивом.

int *arr = malloc(10 * sizeof(int))

В программе на языке C память представляет собой не один непрерывный фрагмент подряд идущих байт. Каждая программа использует 4 типа памяти:

  • CODE
  • STACK
  • DATA
  • HEAP

Фрагмент памяти CODE содержит машинные команды в двоичном коде, которые должен выполнить процессор.

Фрагмент памяти STACK содержит локальные переменные функции и адрес возврата из функции. В каждый момент времени в программе выполняется какая-либо функция. Функция может вызывать другую, функцию. При вызове функции в стеке сохраняются значения всех ее локальных переменных и адрес возврата из функции. При возврате из функции из стека удаляются все ее локальные переменные и удаляется адрес вовзрата.

Фрагмент памяти HEAP служит для динамического выделения памяти из этого фрагмента. Любая память выделенная в куче останется занятой, пока существует сам фрагмент памяти HEAP. Любые переменные, созданные в куче, могут передаваться между функциями с помощью указателей. Для работы с памятью в куче используются функции malloc(), calloc(), realloc(), free().

Если выделить область памяти и не освободить ее, то произойдет утечка памяти, когда эту память уже нельзя переиспользовать.

При обращении к памяти, которую заранее не выделили для себя, получаем ошибку segmentation fault.

Многомерные динамические массивы

Одномерный статический массив:

int a[10]

Одномерный динамический массив:

int *a = malloc(10 * sizeof(int))

Многомерный статический массив:

int a[10][10]

Многомерный динамический массив:

// выделение памяти
int **a = malloc(10 * sizeof(int));
for (int i = 0; i < 10; i++) {
  a[i] = malloc(10 * sizeof(int));
}

// освобождение памяти
for (int i = 0; i < 10; i++) {
  free(a[i]);
}
free(a);

Напишем функцию tr(), которая принимает матрицу размером n x n в виде двумерного динамического массива и транспонирует матрицу.

void tr(int n, int **a) {
  for (int i = 0; i < n; i++) {
    for (int j = i; j < n; j++) {
      int tmp = a[i][j]; // *(j + *(a+i))
      a[i][j] = a[j][i];
      a[j][i] = tmp;
    }
  }
}

9 Работа с числами с плавающей точкой в С

Стандарт IEEE 754 (Institute of Electronical and Electrotechnical Engineers) описывает представление вещественных чисел в памяти компьютера.

Перевод дробного двоичного числа в десятичное

Целая часть представляется положительными степенями двойки, начиная с нуля. Дробная часть (после десятичной точки) представляется отрицательными степенями двойки начиная с -1.

Например, двоичное число 0.1111 переводится в десятичное как

0.1111 = 1/2 + 1/4 + 1/8 + 1/16 = 0.5 + 0.25 + 0.125 + 0.0675

Перевод дробного десятичного числа в двоичного

При преобразовании вещественного десятичного числа в двоичную, целая часть числа преобразовывается отдельно и независимо от дробной части.

Переведем вещественное число 3 / 8 = 0.375 в двоичную систему счисления.

0.375 = 0.25 + 0.125 = 0.011

Этот же результат получается если складывать дроби.

3/8 = 1/4 + 1/8 = 0.011

Третий способ. Умножим дробную часть числа на 2 и получим целую и дробную часть в ответе. Будем записывать число 0 или 1, которое получится в целой части. С дробной частью будем повторять те же действия пока она не станет равна 0

0.375 * 2 = 0.75
0.75 * 2 = 1.5
0.5 * 2 = 1.0
ответ 0.011

Не все числа можно представить в виде конечной дроби. Например, попробуем перевести число 0.1 в двоичную систему.

0.1 * 2 = 0.2
0.2 * 2 = 0.4
0.4 * 2 = 0.8
0.8 * 2 = 1.6
0.6 * 2 = 1.2
ответ 0.0001100110011...

Для хранения целого числа используется 32 бита, что может вместить диапазон от -231 до 231-1.

Как нам хранить в 32 битах вещественные числа если даже в диапазоне от 0 до 1 их бесконечно много.

Нормализованным числом называется число вида a.bbb * 10x. Где a это любая цифра кроме 0, b это любая цифра, x это степень системы счисления, в котором представлено число, в данном случае десятичная система счисления.

В двоичной системе счисления нормализованное число всегда содержит цифру 1 перед десятичной точкой. Поэтому, чтобы не занимать лишнее место в памяти, эту 1 не хранят, а записывают только цифры после десятичной точки.

Денормализованным числом называется число вида 0.bbb * 10x

Число 123.123 запишем в нормализованном виде 1.23123*102. Теперь перепишем число так, чтобы десятичная точка передвинулась и от числа осталась только целая часть 123123*105. При этом мы помним, что в нормализованном числе первая цифра стоит перед точкой, а остальные цифры после точки. Целое число 123123 называется мантиссой, а степень называется экспонента. В памяти, отведенном для хранения вещественного числа, будем хранить отдельно два целых числа - мантиссу и экспоненту.

Точность представления вещественного числа

Точность, с которой можно хранить вещественное число, может быть half (половинная), single (одинарная), double (двойная), quadrple (четверная).

В памяти 1 бит используется для знака, несколько бит для экспоненты, а все оставшиеся биты для мантисы.

  • half. 16 бит. экспонента 5 бит. мантисса 10 бит
  • single. 32 бита. экспонента 8 бит. мантисса 23 бита
  • double. 64 бита. экспонента 11 бит. мантисса 52 бита
  • quadrple. 128 бит. экспонента 15 бит. мантисса 112 бит

Вещественные числа в языках C/C++

В языках C/C++ для работы с вещественными числами используются типы:

  1. float (одинарная точность)
  2. double (двойная точность)
  3. long double (четверная точность)

Число бит для хранения экспоненты влияет на диапазон хранимых чисел.

Число бит для хранения мантисы влияет на точность представления числа.

  • single. Максимальное число 1038. Минимальное число 10-38.
  • double. Максимальное число 10307. Минимальное число 10-307.

Как представить вечественное число 0? Есть два способа

00000000 = 0 10000000 = -0

Когда появляется величина -0? Отрицательное число настолько маленькое, что не входит в диапазон вещественных чисел, записывается как -0. При этом значение этой величины теряется, но остается информация о ее знаке.

Важные уточнения

Есть 4 важных уточнения. Первое касается представления положителной и отрицательной экспоненты. Второе касается записи чисел близких к нулю. Третье касается представления бесконечности в двоичном виде. Четвертое касается записи значения NaN в вещественном числе.

Пусть для экспоненты мы используем p битов. Тогда значение экспоненты может быть от 0 до 2p-1. Экспонента может быть как положительная так и отрицательная. Договорились не хранить экспоненту в двоичном дополнительном коде, поэтому 1 бит под знак не выделяется. Вместо этого выбирается среднее значение равное 2p-1-1. Из экспоненты вычитается среднее значение и получается либо положительное либо отрицательное число.

Мы можем хранить только конечное количество чисел. Причем в диапазоне от 0 до 1 количество чисел меньше чем в диапазоне от 1 до 2 или другом диапазоне между двумя соседними целыми числами. Мы хотим чтобы точность представления числа была одинакова во всех диапазонах. Эту проблему решили следующим образом. Если все числа в записи экспоненты равны 0, то число хранится в денормализованном виде.

Теперь у нас есть максимальное число, которое хранится в денормализованном виде это все 1 в мантисе. И есть минимальное число, которое хранится в нормализованном виде. Этот трюк уплотняет числа, которые находятся около 0.

Когда мы работаем с числами порядка 10-320, то все разряды экспоненты равны 0. При этом мы уже работаем с денормализованным представлением чисел.

Если все биты экспоненты равны 1, а все биты матнисы равны 0, то это является представлением бесконечности. Поскольку знаковый бит определяет является ли число положительным или отрицательным, то можно записать положительную бесконечность и отрицательную бесконечность.

Как можно получить значения бесконечности. Если число настолько сильно удалено от 0, что не помещается в диапазон представимых чисел, то оно считается равным бесконечности.

Если все биты экспоненты равны 1, и не все биты матнисы равны 0, то все полученные значения отведены под значение NaN (not a number). При сложении минус бесконечности и плюс бесконечности получается NaN.

Погрешность преобразования десятичного числа в двоичное

Между любыми двумя вещественными числами, записанными в переменную double находится бесконечное множество чисел, которые невозможно записать в переменную double.

Например, запись double x = 0.1, запишет в переменную x не число 0.1, а число максимально близкое к 0.1, которое можно записать в переменную double.

При различных вычислениях, которые приводят к одному и тому же вещественному числу, можем получить разные числа. Например, получив число 0.1 приближаясь к нему справа, будет записано число максимально близкое к 0.1, но больше него. А при приближении к этому числу слева, будет записано число максимально близкое к 0.1, но меньше него.

Погрешность вычисления вещественных чисел

Пусть мы хотим найти разность двух чисел.

8 - 10-17 = 8

Почему так происходит? Потому что эти два числа нельзя поместить в одной мантисе при одинаковой экспоненте. Компьютер округляет результат до 8.

Погрешность вычисления появляется, когда нельзя записать результат в одной мантисе.

Сравнение вещественных чисел

Из-за погрешности преобразования, два одинаковвых числа нельзя сравнивать операторами сравнения. Их нужно сравнивать, используя разность между ними и сравнивать эту разность с некоторой маленькой величиной, называемой погрешностью.

Ошибка при работе с типами int и double

Попробуем разделить целое число на целое число и результат положить в тип double.

int a = 1;
int b = 2;
double c = a/b; // 0, а не 0.5

При делении целого числа 1 на целое число 2 получается целое число 0, а остаток 0.5 отбрасывается. После этого мы записываем в переменную типа double целое число 0.

Чтобы исправить эту ошибку, при вычислении выражения одна из переменных должна быть типа double. Если в выражении участвует переменная типа double, то результат выражения будет типа double.

int a = 1;
int b = 2;
double c = 1.0*a/b; // 0.5

Напишем функцию print(), которая принимает число типа double и выводит все его биты.

#include 

void print(double d) {
  long long *t = (long long *)&d;

  // вывод мантисы
  for (int i = 0; i < 52; i++) {
    printf("%d", (*t >> i) & 1);
  }
  printf(" ");

  // вывод экспоненты
  for (int i = 52; i < 63; i++) {
    printf("%d", (*t >> i) & 1);
  }
  printf(" ");

  // вывод знакового бита
  printf("%d", (*t >> 63) & 1);
  printf("\n");
  printf("%lf\n", d);
}

int main() {
  printf("Enter 1 double number\n");
  double t;
  scanf("%lf", &t);
  print(t);

  printf("Enter 2 double number for sum\n");
  double t1, t2;
  scanf("%lf %lf", &t1, &t2);
  t = t1 + t2;
  print(t);

  return 0;
}

Прототипы функций

До того как функция будет вызвана, она должна быть объявлена в коде выше. Если есть две функции, которые вызывают друг друга, то в каком порядке мы бы их не записали, все равно компилятор попытается вызвать еще не объявленную функцию и выдаст ошибку. Прототипы функций решают эту проблему.

Прототипом функции называется объявление функции, но без тела функции. Прототип оканчивается точкой с запятой.

int sum(int a, int b);

Разделение программы на несколько файлов

Большие программы обычно не пишут в одном файле и в одной функции main(). В основном файле пишут функцию main(), а другие функции пишут в других файлах.

Заголовочные файлы

Прототипы функций пишут в заголовочных файлах. Заголовочные файлы подключают там, где нужно использовать эти функции.

Файл print.h

void print(int *a, int n);

Файл print.c

#include 

void print(int *a, int n) {
  for (int i = 0; i < n; i++) {
    printf("%d\n", a[i]);
  }
}

Файл main.c

#include "print.h"

int main() {
  int a[5];
  for (int i = 0; i < 5; i++) {
    a[i] = i;
  }
  print(a, 5);

  return 0;
}
