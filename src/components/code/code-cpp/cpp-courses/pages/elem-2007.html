Урок 1. Вводный

Напишем первую программу

#include <iostream>

using namespace std;

int main() {
  cout << "Hello World!";
}

Если консоль сама закрывается, то можно использовать несколько методов, чтобы оставить ее открытой до нажатия на клавишу.

Системная команда PAUSE из библиотеки iostream для задержки консольного окна.

#include <iostream>

int main() {
  cout << "Hello World!";
  system("PAUSE");
}

В результате октроется консоль и мы увидим фразу

Для продолжения нажмите любую клавишу . . .

Консоль будет ожидать нажатия клавиши, прежде чем закрыться.

Можно задержать консоль и без вывода этой фразы.

#include <iostream>

int main() {
  cout << "Hello World!";
  system("PAUSE >> void");
}

Функция getchar() из библиотеки stdio.h для задержки консольного окна

#include <iostream>
#include <stdio.h>

using namespace std;

int main() {
  cout << "Hello World!";
  getchar();
}

Функция getch() из библиотеки conio.h для задержки консольного окна

#include <iostream>
#include <conio.h>

using namespace std;

int main() {
  cout << "Hello World!";
  getch();
}

Урок 2. Переменные

Объявление переменной и инициализация переменной

int x; x = 5;

Можно в одной строчке объявить и инициализировать переменную

int x = 5;

Выведем значение переменной, используя функцию printf из библиотеки stdio.h

#include <stdio.h>

using namespace std;

int main() {
  int x = 5;
  printf("%i", x);
}

Добавим еще одну переменную и выведем ее

#include <stdio.h>

using namespace std;

int main() {
  int x = 5;
  int y = x + 5;
  printf("%i", y);
}

Объявим переменную с плавающей точкой.

#include <stdio.h>

using namespace std;

int main() {
  float x = 5.5;
  printf("%f", x);
}

Объявим переменную двойной точности

#include <iostream>

using namespace std;

int main() {
  double x = 5.56;
  cout << x;
}

Объявим символьную переменную

#include <iostream>

using namespace std;

int main() {
  char x = 'a';
  cout << x;
}

Объявим строковую переменную

#include <iostream>

using namespace std;

int main() {
  string x = "strelok";
  cout << x;
}

Введем число и выведем его

#include <iostream>

using namespace std;

int main() {
  int x;
  cin >> x;
  cout << x;
}

Введем вещественное число и выведем его

#include <iostream>

using namespace std;

int main() {
  double x;
  cin >> x;
  cout << x;
}

Введем символ и выведем его

#include <iostream>

using namespace std;

int main() {
  char x;
  cin >> x;
  cout << x;
}

Введем строку и выведем ее. Функция cin запишет в строковую переменную строку только до первого пробельного символа.

#include <iostream>

using namespace std;

int main() {
  string x;
  cin >> x;
  cout << x;
}

Введем строку до конца, вместе с пробельными символами. Функция getline() принимает два аргумента, первый это поток cin откуда брать строку, а второй это переменная x куда записывать строку.

#include <iostream>

using namespace std;

int main() {
  string x;
  getline(cin, x);
  cout << x;
}

Урок 3. Логика

Тип данных bool содержит переменную логического типа, принимающую одно из двух значений true или false. Выражения, возвращающие логические значения используются в условных операторах.

#include <iostream>

using namespace std;

int main() {
  int x = 45;
  if (x < 50) {
    cout << "x < 50";
  } else if (x > 50) {
    cout << "x > 50";
  } else {
    cout << "x = 50";
  }
}

Урок 4. Циклы

Напишем цикл while

#include <iostream>

using namespace std;

int main() {
  int n = 5;

  while (n != 0) {
    cout << "n = " << n << endl;
    n--;
  }
}

Напишем цикл do while

#include <iostream>

using namespace std;

int main() {
  int n = 5;

  do {
    cout << "n = " << n << endl;
    n--;
  } while (n != 0);
}

Напишем цикл for

#include <iostream>

using namespace std;

int main() {
  for (int i = 0; i < 5; i++) {
    cout << "i = " << i << endl;
  }
}

Урок 5. Массивы

Создадим массив из пяти нулей.

#include <iostream>

using namespace std;

int main() {
  int x[5] = {0};
  x[2] = 10;
  for (int i = 0; i < 5; i++) {
    cout << x[i] << endl;
  }
}

Создадим массив из простых чисел.

#include <iostream>
#include <iomanip>

using namespace std;

int main() {
  const int size = 1000;
  int A[size] = {};
  for (int i = 2; i < size; i++) {
    if(A[i] == 0) {
      for (int j = i; j < size; j += i) {
        A[i] = 2;
      }
      A[i] = 1;
    }
  }
  int cnt = 0;
  for (int i = 0; i < size; i++) {
    if (A[i] == 1) {
      cout << setw(6) << "   ";
      cnt++;
      if (cnt % 5) {
        cout << endl;
      }
    }
  }
}
