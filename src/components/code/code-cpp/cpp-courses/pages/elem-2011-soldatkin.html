3 Условный оператор if

Введем целое число и проверим отрицательное оно или положительное.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter number ";
  int a;
  cin >> a;

  if (a < 0) {
    cout << a << " < " << 0;
  } else if (a > 0) {
    cout << a << " > " << 0;
  } else if (a == 0) {
    cout << a << " = " << 0;
  }
}

Введем целое число и проверим находится ли оно в диапазоне от 0 до 10.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter number ";
  int a;
  cin >> a;

  if (a >= 0 && a <= 10) {
    cout << a << " in range 0 - 10 " << endl;
  } else if (a < 0 || a > 10) {
    cout << a << " not in range 0 - 10 " << endl;
  }
}

4 Оператор switch

Рассмотрим операции +, -, *, /, % с целыми числами. Операции можно выбирать с помощью пунктов меню.

#include <iostream>

using namespace std;

int main() {
  int a, b, c;
  int item;

  // вводим два числа
  cout << "Enter number 1 ";
  cin >> a;
  cout << "Enter number 2 ";
  cin >> b;

  // показываем меню
  cout << "1. + \n";
  cout << "2. - \n";
  cout << "3. * \n";
  cout << "4. / \n";
  cout << "5. % \n";

  // выбираем пункт меню
  cout << "Enter item ";
  cin >> item;

  switch (item) {

    case 1:
      c = a + b;
      cout << a << " + " << b
          << " = " << c << endl;
      break;

    case 2:
      c = a - b;
      cout << a << " - " << b
          << " = " << c << endl;
      break;

    case 3:
      c = a * b;
      cout << a << " * " << b
          << " = " << c << endl;
      break;

    case 4:
      if (b != 0) {
        c = a / b;
        cout << a << " / " << b
            << " = " << c << endl;
      } else {
        cout << "Error: division by zero" << endl;
      }
      break;

    case 5:
      if (b != 0) {
        c = a % b;
        cout << a << " % " << b
            << " = " << c << endl;
      } else {
        cout << "Error: division by zero" << endl;
      }
      break;

    default:
      cout << "error";
      break;

  }
}

Рассмотрим операции +, -, *, /, % с целыми числами. Операции можно выбирать с помощью пунктов меню. Добавим пункт выхода из меню. Программа будет выполняться пока не будет выбран пункт выхода.

            << " = " << c << endl;
        break;

      case 2:
        c = a - b;
        cout << a << " - " << b
            << " = " << c << endl;
        break;

      case 3:
        c = a * b;
        cout << a << " * " << b
            << " = " << c << endl;
        break;

      case 4:
        if (b != 0) {
          c = a / b;
          cout << a << " / " << b
              << " = " << c << endl;
        } else {
          cout << "Error: division by zero" << endl;
        }
        break;

      case 5:
        if (b != 0) {
          c = a % b;
          cout << a << " % " << b
              << " = " << c << endl;
        } else {
          cout << "Error: division by zero" << endl;

6 Цикл for

#include <iostream>

int main() {

7 Календарь 2026

Напишем календарь на 2026 год. 1 января это четверг.

#include <iostream>

using namespace std;

int main() {
  int days; // число дней в месяце
  int d = 4; // день недели. 1 января это 4 день

  for (int month = 1; month < 13; month++) {
    switch (month) {

      case 1:
        cout << "\t\t\tJanuary\n";
        days = 31;
        break;

      case 2:
        cout << "\t\t\tFebruary\n";
        days = 28;
        break;

      case 3:
        cout << "\t\t\tMarch\n";
        days = 31;
        break;

      case 4:
        cout << "\t\t\tApril\n";
        days = 30;
        break;

      case 5:
        cout << "\t\t\tMay\n";
        days = 31;
        break;

      case 6:
        cout << "\t\t\tJune\n";
        days = 30;
        break;

      case 7:
        cout << "\t\t\tJuly\n";
        days = 31;
        break;

      case 8:
        cout << "\t\t\tAugust\n";
        days = 31;
        break;

      case 9:
        cout << "\t\t\tSeptember\n";
        days = 30;
        break;

      case 10:
        cout << "\t\t\tOktober\n";
        days = 31;
        break;

      case 11:
        cout << "\t\t\tNovember\n";
        days = 30;
        break;

      case 12:
        cout << "\t\t\tDecember\n";
        days = 31;
        break;
    }

    cout << "Mo\tTu\tWe\tTh\tFr\tSa\tSu\n";

    // выставим табуляцию до 1 января
    for (int tab = 1; tab < d; tab++) {
      cout << '\t';
    }

    // заполним дни месяца
    for (int day = 1; day <= days; day++) {
      cout << day << '\t';
      if (d % 7 == 0) {
        cout << '\n';
        d = 0;
      }
      d++;
    }

    // переход на следуюдий месяц
    cout << '\n';
  }
}

8 Русские буквы в консоли

Буквы английского алфавита распологаются в первой части таблицы ASCII. Буквы русского алфавита распологаются во второй части таблицы ASCII.

Таблица ASCII это кодировка символов, где каждый символ занимает место в памяти 1 байт. Каждый символ кодируется 8 битами. Символы с кодами от 0 до 127 - это первая часть таблицы, которая не меняется. Символы с кодами от 128 до 255 - это вторая часть таблицы, в которую можно поместить национальный алфавит. Для этого нужно применить нужную кодировку символов.

Попробуем вывести в консоль строку из английский символов и сторку из русских символов.

#include <iostream>

using namespace std;

int main() {
  cout << "Hello" << '\n';
  cout << "Привет" << '\n';
}

Английский текст выводится правильно

Hello

Русский текст не выводится

╨Я╤А╨╕╨▓╨╡╤В

Запишем текст в переменные типа string и попробуем вывести их в консоль снова.

#include <iostream>

using namespace std;

int main() {
  string eng = "Hello";
  string rus = "Привет";

  cout << eng << '\n';
  cout << rus << '\n';
}

Английский текст выводится правильно

Hello

Русский текст не выводится

╨Я╤А╨╕╨▓╨╡╤В

Введем с консоли имя и запишем его в переменную типа string и попробуем вывести его в консоль снова.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter name ";
  string name;
  cin >> name;

  cout << name << '\n';
}

Теперь любые символы и английские и русские правильно отображаются на экране. Почему так происходит? Потому что когда мы вводим русский символ в консоли, то мы его получаем и записываем в переменную в той кодировке, в какой он дальше будет выведен обратно на консоль. А когда мы пишем русский символ внутри программы, то он будет записан в переменную в той кодировке, в которой работает редактор кода. И эта кодировка может отличаться от кодировки символов консоли.

По умолчанию в консоли используется кодировка WINDOWS-866. Сохраним текст программф в кодировке WINDOWS-866.

#include <iostream>

using namespace std;

int main() {
  string eng = "Hello";
  string rus = "Привет";

  cout << eng << '\n';
  cout << rus << '\n';
}

Русский текст теперь должен отображаться правильно.

Привет

Кодировку консоли можно изменить и внутри программы

system("chcp 866"); // кодировка WINDOWS-866 system("chcp 1251"); // кодировка WINDOWS-1251

Сохраним текст программы в кодировке WINDOWS-866. Изменим внутри программы кодировку консоли на WINDOWS-866

#include <iostream>

int main() {

  string rus = "Привет";

  cout << rus << '\n';
}

Теперь русский текст выводится правильно.

Заголовочный файл windows.h содержит функцию SetConsoleCP() которая также меняет кодировку консоли. Напишем программу, которая вводит русский текст с консоли в кодировке 1251, а выводит текст на косноль в кодировке 866.

#include <iostream>
#include <windows.h>

using namespace std;

int main() {
  string rus;

  SetConsoleCP(1251);
  cin >> rus;

  SetConsoleCP(866);
  cout << rus << '\n';
}
