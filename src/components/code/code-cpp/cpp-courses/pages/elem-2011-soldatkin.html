3 Условный оператор if

Введем целое число и проверим отрицательное оно или положительное.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter number ";
  int a;
  cin >> a;

  if (a < 0) {
    cout << a << " < " << 0;
  } else if (a > 0) {
    cout << a << " > " << 0;
  } else if (a == 0) {
    cout << a << " = " << 0;
  }
}

Введем целое число и проверим находится ли оно в диапазоне от 0 до 10.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter number ";
  int a;
  cin >> a;

  if (a >= 0 && a <= 10) {
    cout << a << " in range 0 - 10 " << endl;
  } else if (a < 0 || a > 10) {
    cout << a << " not in range 0 - 10 " << endl;
  }
}

4 Оператор switch

Рассмотрим операции +, -, *, /, % с целыми числами. Операции можно выбирать с помощью пунктов меню.

#include <iostream>

using namespace std;

int main() {
  int a, b, c;
  int item;

  // вводим два числа
  cout << "Enter number 1 ";
  cin >> a;
  cout << "Enter number 2 ";
  cin >> b;

  // показываем меню
  cout << "1. + \n";
  cout << "2. - \n";
  cout << "3. * \n";
  cout << "4. / \n";
  cout << "5. % \n";

  // выбираем пункт меню
  cout << "Enter item ";
  cin >> item;

  switch (item) {

    case 1:
      c = a + b;
      cout << a << " + " << b
          << " = " << c << endl;
      break;

    case 2:
      c = a - b;
      cout << a << " - " << b
          << " = " << c << endl;
      break;

    case 3:
      c = a * b;
      cout << a << " * " << b
          << " = " << c << endl;
      break;

    case 4:
      if (b != 0) {
        c = a / b;
        cout << a << " / " << b
            << " = " << c << endl;
      } else {
        cout << "Error: division by zero" << endl;
      }
      break;

    case 5:
      if (b != 0) {
        c = a % b;
        cout << a << " % " << b
            << " = " << c << endl;
      } else {
        cout << "Error: division by zero" << endl;
      }
      break;

    default:
      cout << "error";
      break;

  }
}

7 Календарь 2026

Напишем календарь на 2026 год. 1 января это четверг.

#include <iostream>

using namespace std;

int main() {
  int days; // число дней в месяце
  int d = 4; // день недели. 1 января это 4 день

  for (int month = 1; month < 13; month++) {
    switch (month) {

      case 1:
        cout << "\t\t\tJanuary\n";
        days = 31;
        break;

      case 2:
        cout << "\t\t\tFebruary\n";
        days = 28;
        break;

      case 3:
        cout << "\t\t\tMarch\n";
        days = 31;
        break;

      case 4:
        cout << "\t\t\tApril\n";
        days = 30;
        break;

      case 5:
        cout << "\t\t\tMay\n";
        days = 31;
        break;

      case 6:
        cout << "\t\t\tJune\n";
        days = 30;
        break;

      case 7:
        cout << "\t\t\tJuly\n";
        days = 31;
        break;

      case 8:
        cout << "\t\t\tAugust\n";
        days = 31;
        break;

      case 9:
        cout << "\t\t\tSeptember\n";
        days = 30;
        break;

      case 10:
        cout << "\t\t\tOktober\n";
        days = 31;
        break;

      case 11:
        cout << "\t\t\tNovember\n";
        days = 30;
        break;

      case 12:
        cout << "\t\t\tDecember\n";
        days = 31;
        break;
    }

    cout << "Mo\tTu\tWe\tTh\tFr\tSa\tSu\n";

    // выставим табуляцию до 1 января
    for (int tab = 1; tab < d; tab++) {
      cout << '\t';
    }

    // заполним дни месяца
    for (int day = 1; day <= days; day++) {
      cout << day << '\t';
      if (d % 7 == 0) {
        cout << '\n';
        d = 0;
      }
      d++;
    }

    // переход на следуюдий месяц
    cout << '\n';
  }
}

8 Русские буквы в консоли

Буквы английского алфавита распологаются в первой части таблицы ASCII. Буквы русского алфавита распологаются во второй части таблицы ASCII.

Таблица ASCII это кодировка символов, где каждый символ занимает место в памяти 1 байт. Каждый символ кодируется 8 битами. Символы с кодами от 0 до 127 - это первая часть таблицы, которая не меняется. Символы с кодами от 128 до 255 - это вторая часть таблицы, в которую можно поместить национальный алфавит. Для этого нужно применить нужную кодировку символов.

Попробуем вывести в консоль строку из английский символов и сторку из русских символов.

#include <iostream>

using namespace std;

int main() {
  cout << "Hello" << '\n';
  cout << "Привет" << '\n';
}

Английский текст выводится правильно

Hello

Русский текст не выводится

╨Я╤А╨╕╨▓╨╡╤В

Запишем текст в переменные типа string и попробуем вывести их в консоль снова.

#include <iostream>

using namespace std;

int main() {
  string eng = "Hello";
  string rus = "Привет";

  cout << eng << '\n';
  cout << rus << '\n';
}

Английский текст выводится правильно

Hello

Русский текст не выводится

╨Я╤А╨╕╨▓╨╡╤В

Введем с консоли имя и запишем его в переменную типа string и попробуем вывести его в консоль снова.

#include <iostream>

using namespace std;

int main() {
  cout << "Enter name ";
  string name;
  cin >> name;

  cout << name << '\n';
}

Теперь любые символы и английские и русские правильно отображаются на экране. Почему так происходит? Потому что когда мы вводим русский символ в консоли, то мы его получаем и записываем в переменную в той кодировке, в какой он дальше будет выведен обратно на консоль. А когда мы пишем русский символ внутри программы, то он будет записан в переменную в той кодировке, в которой работает редактор кода. И эта кодировка может отличаться от кодировки символов консоли.

По умолчанию в консоли используется кодировка WINDOWS-866. Сохраним текст программф в кодировке WINDOWS-866.

#include <iostream>

using namespace std;

int main() {
  string eng = "Hello";
  string rus = "Привет";

  cout << eng << '\n';
  cout << rus << '\n';
}

Русский текст теперь должен отображаться правильно.

Привет

Размер массива должен быть известен до компиляции. Определим его как константу. Элементы массивы можно задавать при его объявлении. Размер массива можно не указывать, если его элементы сразу получают значения. В этом случае его размер определяется количеством созданных элементов. Его размер можно определить оператором sizeof.

#include <iostream>

using namespace std;

int main() {
  const int n = 5;
  int mas[n] = {315, 41, 8, 62, 97};
  for (int i = 0; i < 5; i++) {
    cout << mas[i] << ' ';
  }

  cout << '\n';

  int mas2[] = {84, 37, 90};
  int m = sizeof(mas2) / sizeof(int);
  for (int i = 0; i < m; i++) {
    cout << mas2[i] << ' ';
  }
}

11 Указатели (ч.1)

Указатели и динамическое распределение памяти. Указатели это тип переменных, которые содержат адрес других переменных. Динамическое распределение это выделение памяти под переменные во время выполнения программы, а не во время компиляции.

Создадим переменную a, указатель pa на переменную a, указатель ppa на указатель pa. Посмотрим значения переменных и их адреса.

#include <iostream>

using namespace std;

int main() {
  int a = 40;
  int *pa = &a;
  int **ppa = &pa;

  cout << "a = " << a << '\n';
  cout << "&a = " << &a << '\n';
  cout << '\n';
  cout << "*pa = " << *pa << '\n';
  cout << "pa = " << pa << '\n';
  cout << "&pa = " << &pa << '\n';
  cout << '\n';
  cout << "**ppa = " << **ppa << '\n';
  cout << "*ppa = " << *ppa << '\n';
  cout << "ppa = " << ppa << '\n';
  cout << "&ppa = " << &ppa << '\n';
}

Создадим двумерный массив и заполним его случайными числами.

    for (int j = 0; j < m; j++) {
      mas[i][j] = rand() % 100;
      cout << mas[i][j] << '\t';
    }
    cout << '\n';
  }
}

С указателя можно работать с двумерным массивом как с одномерным.

#include <iostream>

using namespace std;

int main() {

  int m = 3;
  int mas[n][m];

11 Указатели (ч.2)

До этого мы работали со статическими массивами. Теперь создадим динамический массив. Если в программе динамически выделяется память с помощью оператора new, то потом эту память нужно обязательно освободить с помощью оператора delete.

#include <iostream>

using namespace std;

int main() {
  int n = 10;

  int *mas = new int[n];

  for (int i = 0; i < n; i++) {
    mas[i] = rand() % 100;
    cout << mas[i] << '\t';
  }

  delete [] mas;
}

При создании динамического массива, его размер может быть задан во время выполнения программы. При создании статического массива, его размер должен быть известен во время компиляции.

#include <iostream>

using namespace std;

int main() {
  int n;
  cout << "Enter array size ";
  cin >> n;

  int *mas = new int[n];

  for (int i = 0; i < n; i++) {
    mas[i] = rand() % 100;
    cout << mas[i] << '\t';
  }

  delete [] mas;
}

Создадим двумерный динамический массив. Сначала создадим массив указателей. Затем в цикле для каждого указателя выделем память для одномерного массива.

#include <iostream>

using namespace std;

int main() {
  int n;
  int m;
  cout << "Enter array rows ";
  cin >> n;
  cout << "Enter array cols ";
  cin >> m;

  int **mas = new int*[n];

  for (int i = 0; i < n; i++) {
    mas[i] = new int[m];
  }

  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
      mas[i][j] = rand() % 200;
      cout << mas[i][j] << '\t';
    }
    cout << '\n';
  }

  for (int i = 0; i < n; i++) {
    delete [] mas[i];
  }

  delete [] mas;
}

12 Строки

Библиотека cstring содержит функцию strcmp(), которая сравнивает две строки. Возвращает 0 если строки равны, возвращает 1 если первая строка больше, возвращает -1 если первая строка меньше.

#include <iostream>
#include <cstring>

using namespace std;

int main() {
  char str1[] = "dima";
  char str2[] = "dima";
  int res = strcmp(str1, str2);

  cout << res;
}

Функция stricmp сравнивает две строки без учета регистра.

#include <iostream>

Введем строку с клавиатуры. Для ввода имени из четырех букв нам нужен массив из 5 сивовол, потому что последний символ это всегда символ конца строки, который записывается в массив автоматически.

Если ввести 5 символов и попытаться записать их в строку из 5 символов, то получим ошибку. Чтобы этого не произошло будем использовать функцию getline, которая запишет в переменную name число символов, равное размеру переменной name.

Как записать в массив текст с консоли когда мы заранее не знаем количество символов. Это можно сделать с помощью буферизированного потока. Для начала посмотрим как работать с буферизированным потоком.

#include <iostream>
#include <strstream>

using namespace std;

int main() {
  ostrstream ostr; // создадим буферизированный поток

  ostr << ends; // запишем символ конца строки
  char *p;
  p = ostr.str();

  cout << p;
}

Теперь запишем в массив текст с консоли когда мы заранее не знаем количество символов. С указателем на строку можно работать как с обычным массивом.

#include <iostream>

  } while (ch != '\n');

  ostr << ends;
  p = ostr.str();
  cout << "\n\n";
  cout << p;

  if (p[0] && p[1] && p[2]) {
    cout << "1 symbol " << p[0] << '\n';
    cout << "2 symbol " << p[1] << '\n';
    cout << "3 symbol " << p[2] << '\n';
  }
}

Напишем функцию, которая заполняет массив случайными числами. Имя массива одновременно является адресом на его первый элемент.

#include <iostream>

Перегрузка функции это когда функции имеют одно и то же имя, но принимают разные параметры. Создадим перегруженную функцию init для двумерного массива. Создадим перегруженную функцию show для двумерного массива.

#include <iostream>

using namespace std;

void show(int *arr, int n) {
  for (int i = 0; i < n; i++) {
    cout << arr[i] << ' ';
  }
  cout << '\n';
}

void show(int *arr, int n, int m) {
  for (int i = 0; i < n * m; i++) {
    cout << arr[i] << ' ';
    if ((i+1) % m == 0) {
      cout << '\n';
    }
  }
  cout << '\n';
}

void init(int *arr, int n) {
  for (int i = 0; i < n; i++) {
    arr[i] = rand() % 50;
  }
}

void init(int *arr, int n, int m) {
  for (int i = 0; i < n * m; i++) {
    arr[i] = rand() % 50;
  }
}

int main() {
  const int n = 5;
  const int m = 10;
  int mas[n];

  init(&mas[0], n);
  show(&mas[0], n);
  show(mas, n);

  cout << '\n';

  int mas2[n][m];
  int *p = &mas2[0][0]; // указатель на двумерный массив

  init(&mas2[0][0], n, m);
  show(&mas2[0][0], n, m);
  show(p, n, m);
}
