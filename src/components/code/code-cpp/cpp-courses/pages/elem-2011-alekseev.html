1 Таблица кодировки

Создадим таблицу кодировки с помощью ООП. Создадим класс CodeTable который хранит массив символов array. Поля lbound и hbound описывают нижнюю и верхнюю границу диапазона символов. В конструкторе задаются lbound и hbound и происходит инициализация таблицы преобразования массивом iniTable. Второй конструктор это конструктор копирования. Деструктор нужен для того чтобы освободить память от массива array. Определение оператора [] индексации.

#pragma once это защита от повторного включения заголовочного файла.

Файл codetable.h

#pragma once

typedef unsigned char byte;

class CodeTable {
  byte lBound;
  byte hBound;
  byte *array;

public:
  // конструктор
  CodeTable(byte low, byte high, byte* iniTable = 0);

  // конструктор копирования
  CodeTable(const CodeTable &src);

  // деструктор
  ~CodeTable();

  byte& operator[](byte i);
};

В конструкторе сначала проверяем что верхняя граница не ниже нижней границы. Потом выделяем память под массив array. В каждую ячейку массива записываем его индекс. Теперь таблица кодов будет преобразовываться код сам в себя. Далее проверяется условие if (iniTable). Если массив iniTable не задан, то условие if не выполняется и инициализация таблицы не выполняется. Иначе инициализируем массив array массивом iniTable.

В конструкторе копирования выделяется память под массив и в него копируются данные из переданного массива.

Деструктор освобождает память, которую коснтруктор динамически выделял под массив array с помощью оператора new.

Определение оператора [] - это операция индексации. Сначала проверяется что индекс попадает в границы между lbound и hbound. Потом из индекса вычитается значение нижней границы чтобы получить индекс элемента в массиве array.

#include <stdexcept> подключение стандартной библиотеки для поддержки типовых исключений.

Файл codetable.cpp

#include <stdexcept>
#include <string>
#include "codetable.h"

using namespace std;

// конструктор
CodeTable::CodeTable(byte low, byte high, byte* iniTable) {
  if ((hBound = high) < (lBound = low)) {
    throw out_of_range("CodeTable constructor: lower bound is higher than upper one");
  }
  array = new byte[high - low + 1];
  for (byte code = lBound; ; code++) {
    array[code - lBound] = code;
    if (code == hBound) break;
  }

  if (iniTable) {
    for (int ndx = 0; ndx <= high - lBound && iniTable[ndx]; ndx++) {
      array[ndx] = iniTable[ndx];
    }
  }
}

// конструктор копирования
CodeTable::CodeTable(const CodeTable &src) {
  array = new byte[(hBound = src.hBound) - (lBound = src.lBound) + 1];

  for (byte ndx = lBound; ; ndx++) {
    array[ndx - lBound] = src.array[ndx - lBound];
    if (ndx == hBound) break;
  }
}

// деструктор
CodeTable::~CodeTable() {
  delete[] array;
}

// операция индексации []
byte & CodeTable:: operator[] (byte i) {
  if (i < lBound || i > hBound) {
    throw out_of_range("Index is out of range");
  }
  return array[i - lBound];
}

Функция doCode заполняет массив dest элементами преобразованными по таблице преобразования codeTable с помощью операции индексации, которая задана в классе CodeTable.

В функции main создадим экземпляр класса CodeTable. В цикле for мы заменяем маленькие буквы алфавита по правилу 'a' заменяется на 'z', 'b' заменяется на 'y' и так далее.

Создадим строку src для преобразования. Создадим строку dst, в которой будем хранить результат преобразования.

Функция memset копирует данные из строки src в строку dst.

Функция doCode преобразовывает символы по таблице символов.

Файл main.cpp

#include <iostream>
#include <string>
#include "codetable.h"

using namespace std;

void doCode(byte* source, byte* dest, CodeTable & codeTable) {
  for (int i = 0; source[i]; i++) {
    dest[i] = codeTable[source[i]];
  }
}

int main() {
  CodeTable codeTable(32, 255);

  for (int i = 0; i < 13; i++) {
    byte b = codeTable['a' + i];
    codeTable['a' + i] = codeTable['z' - i];
    codeTable['z' - i] = b;
  }

  byte *src = (byte*)"Hello, world!";
  byte *dst = new byte[strlen((char*)src) + 1];
  memset(dst, 0, strlen((char*)src) + 1);
  doCode(src, dst, codeTable);
  cout << "Source string : " << src <<'\n';
  cout << "Destination string : " << dst <<'\n';

  return 0;
}

2 Динамический массив

В статическом массиве число элементов изменять нельзя. Мы хотим создать динамический массив в котором можно увеличивать или уменьшать число элементов. Сделаем его на основе статического массива.

Пусть нам нужен динамический массив, который изначально содержит 5 элементов, тогда создадим статичкский массив из 10 элементов. Заведем переменную elementsCount = 5, которая указывает на то что в число элементов в массиве равно пяти, хотя память выделена под 10 элементов.

Добавим в этот массив еще один элемент, тогда elementsCount = 6.

Добавим еще 5 элементов и получится массив из 11 элементов. Но тогда для хранения 11 элементов нам не хватит размера массива из 10 элементов. Тогда создадим новый статический массив на 22 элемента. Скопируем в него 6 элементов из старого массива и добавим 5 новых элементов. Таким образом, в новом массиве будет elementsCount = 11. А старый массив удалим и освободим от него память.

template <class Elem> означает что класс DinamicArray является шаблонным. Это сделано для того чтобы в массиве можно было сохранять элементы любого типа данных. Поле size это количество элементов в массиве. Поле maxSize это максимальное количество элементов массива. Поле array это указатель на сам массив, который хранится в куче.

operator [] оператор индексации нужен чтобы к элементам DynamicArray мы могли обращаться как к обычному массиву через квадратные скобки.

Метод Resize изменяет размер массива.

Метод Add добавляет элемент в массив.

Метод Enlarge увеличивает размер массива.

Метод Shrink уменьшает размер массива.

В условии if ((size = sz) < 0) сначала переменной size присваивается значение переменной sz, а за тем проверяется что это значение меньше нуля. Если size меньше нуля то выбрасывается исключение.

Операция индексации проверяет что индекс не выходит за границы массива.

Метод Resize принимает параметр delta. Если delta больше нуля то вызывается метод Enlarge, если delta меньше нуля то вызывается метод Shrink.

Метод Add добавляет новый элемент в конец массива и увеличивает его размер на 1.

Метод Enlarge создает новый массив, большего размера. В условии if ((size += delta) > maxSize) сначала увеличивается переменная size на значение delta, а затем проверяется что это значение больше maxSize. Если больше, то выделяется место для нового массива и старый массив переносится в новое место.

Метод Shrink уменьшает число элементов size в массиве. Но не уменьшает место, выделенное под массив.

Файл DinamicArray.h

#pragma once

#include <stdexcept>

using namespace std;

template <class Elem>
class DynamicArray {
private:
  int size;
  int maxSize;
  Elem *array;

public:
  // конструктор
  DynamicArray(int sz = 0, int maxSz = 0, Elem *iniArray = NULL);

  // конструктор копирования
  DynamicArray(const DynamicArray<Elem> & a) {
    size = maxSize = 0;
    *this = a;
  }

  // деструктор
  ~DynamicArray() { delete[] array; }

  // оператор индексации
  Elem & operator [] (int i);

  // оператор присваивания
  DynamicArray<Elem> & operator = (const DynamicArray<Elem> & a);
  void Resize(int delta);
  void Add(const Elem & e);

private:
  void Enlarge(int delta);
  void Shrink(int delta);
};

// реализация конструктора класса
template <class Elem>
DynamicArray<Elem>::DynamicArray(int sz, int maxSz, Elem *iniArray) {
  if ((size = sz) < 0) {
    throw out_of_range("DynamicArray constructor : Negative array size");
  }
  maxSize = (maxSz < sz ? sz : maxSz);
  array = new Elem[maxSize];
  if (iniArray) {
    for (int i = 0; i < size; i++) {
      array[i] = iniArray[i];
    }
  }
}

// реализация операции индексации
template <class Elem>
Elem & DynamicArray<Elem>::operator [] (int i) {
  if (i < 0 || i >= size) {
    throw out_of_range("operator [] : Array index is out of range");
  }
  return array[i];
}

// реализация операции присваивания
template <class Elem>
DynamicArray<Elem> & DynamicArray<Elem>::operator=(const DynamicArray<Elem> & a) {
  Resize(a.size - size);
  for (int i = 0; i < size; i++) {
    array[i] = a.array[i];
  }
  return *this;
}

template <class Elem>
void DynamicArray<Elem>::Resize(int delta) {
  if (delta > 0) {
    Enlarge(delta);
  } else if (delta < 0) {
    Shrink(-delta);
  }
}

template <class Elem>
void DynamicArray<Elem>::Add(const Elem & e) {
  Resize(1);
  array[size-1] = e;
}

template <class Elem>
void DynamicArray<Elem>::Enlarge(int delta) {
  if ((size += delta) > maxSize) {
    maxSize = size;
    Elem *newArray = new Elem[maxSize];
    for (int i = 0; i < size - delta; i++) {
      newArray[i] = array[i];
    }
    array = newArray;
  }
}

template <class Elem>
void DynamicArray<Elem>::Shrink(int delta) {
  size = (delta > size ? 0: size- delta);
}

В программе сначала создается объект класса DynamicArray. Заполним этот массив так чтобы элемент i содержал значение i. В цикле от 0 до 13 символы букв английского алфавита от 'a' до 'z' меняются местами по принципу 'a' меняется местами с 'z', 'b' менется местами с 'y' и так далее.

Создаем статическую строку src и строку в динамической памяти dst. Функция memset заполняет строку dst нулями. Функция DoCode производит замену символов строки src согласно таблице codeTable и записывает новую строку в dst.

Файл main.cpp

#include <iostream>
#include "DynamicArray.h"

using namespace std;

void DoCode(char * source, char * dest, DynamicArray<char> & codeTable) {
  for (int i = 0; source[i]; i++) {
    dest[i] = codeTable[source[i]];
  }
}

int main() {
  DynamicArray<char> codeTable(128);
  for (int c = 0; c < 128; c++) codeTable[c] = c;

  for (int i = 0; i < 13; i++) {
    char b = codeTable['a' + i];
    codeTable['a' + i] = codeTable['z' - i];
    codeTable['z' - i] = b;
  }

  char src[] = "Hello, World!";
  char *dst = new char[strlen(src) + 1];

  memset(dst, 0, strlen(src) + 1);

  DoCode(src, dst, codeTable);

  cout << "Source string : " << src << '\n';
  cout << "Destination string : " << dst << '\n';

  return 0;
}

3 Связный список

Связный список это последовательность элементов, каждый из которых хранит значение элемента и указатель на следующий элемент. Последний элемент хранит указатель на NULL. Задается список всегда указателем на его первый элемент.

Двусвязный список это связный список, каждый элемент которого имеет еще один указатель на предыдущий элемент. Первый элемент вторым указателем указывает на NULL.

Чтобы создать список, нужно создать два типа данны. Первый тип данных для элемента списка. Его можно сделать шаблонным, чтобы список мог хранить данные любого типа. Второй тип данных для самого списка, которых хранит указатель на первый элемент списка и логику работы со списком, например, добавление и удаление элемента из списка.

Создадим класс IntList для создания списка из элементов типа int.

Внутри класса создадим структуру ListItem для создания элемента списка. Поле item хранит значение элемента. Указатель next указывает на следующий элемент списка. Конструктор задает начальные значения полям item и next при создании элемента.

Внутри класса имеем три поля. itemsCount это число элементов в списке. Это поле должно обновляться при добавлении или удалении элемента из списка. first это указатель на первый элемент списка. last это указатель на последний элемент списка.

Конструктор по умолчанию создает пустой связный список, где число элементов в списке 0 и указатели fist и last равны NULL.

Конструктор копирования создает полную копию связного списка, который приходит к нему как параметр.

Деструктор удаляет из памяти все элементы связного списка.

Метод Head возвращает значение первого элемента.

Метод Tail возвращает значение последнего элемента.

Метод AddLast принимающий в качестве параметра связный список, добавляет в конец списка другой связный список.

Метод AddFirst добавляет элемент в начало списка.

Метод AddLast принимающий в качестве параметра значение, добавляет элемент с этим значением в конец списка.

Метод RemoveFirst удаляет первый элемент и возвращает его значение.

Метод Remove удаляет элемент со значением n и возвращает true если элемент успешно удален.

Метод Insert вставляет элемент со значением n. Список получается упорядоченным.

Метод GetItemsCount возвращает число элементов.

Метод GetAllItemsInfo возвращает строку с информацией о элементах списка.

Файл IntList.h

#pragma once

#include <string>
#include <stdlib.h>

using namespace std;

class IntList {
private:
  struct ListItem {
    int item;
    ListItem *next;

    // конструктор
    ListItem(int i, ListItem *n = NULL) {
      item = i;
      next = n;
    }
  };

  int itemsCount;
  ListItem *first;
  ListItem *last;

public:
  // конструктор по умолчанию
  IntList() {
    itemsCount = 0;
    first = last = NULL;
  }

  // конструктор копирования
  IntList(const IntList & src);

  // деструктор
  ~IntList();

  int Head() const {
    return first->item;
  }

  int & Head() {
    return first->item;
  }

  int Tail() const {
    return last->item;
  }

  int & Tail() {
    return last->item;
  }

  void AddLast(const IntList & src);
  void AddFirst(int item);
  void AddLast(int item);
  int RemoveFirst();
  bool Remove(int n);
  void Insert(int n);

  int GetItemsCount() {
    return itemsCount;
  }

  string GetAllItemsInfo();
};

Файл IntList.cpp

#include <sstream>
#include <string>
#include "IntList.h"

using namespace std;

// конструктор копирования
IntList::IntList(const IntList & src) {
  itemsCount = 0;
  first = last = NULL;
  AddLast(src);
}

// деструктор
IntList::~IntList() {
  ListItem *current = NULL;
  ListItem *next = first;
  while (next) {
    current = next;
    next = next->next;
    delete current;
  }
}

void IntList::AddLast(const IntList & src) {
  for (ListItem *cur = src.first; cur; cur = cur->next) {
    AddLast(cur->item);
  }
}

void IntList::AddFirst(int item) {
  ListItem *newItem = new ListItem(item, first);
  if (!first) {
    last = newItem;
  }
  first = newItem;
  itemsCount++;
}

void IntList::AddLast(int item) {
  ListItem *newItem = new ListItem(item);
  if (!last) {
    first = newItem;
  } else {
    last->next = newItem;
  }
  last = newItem;
  itemsCount++;
}

int IntList::RemoveFirst() {
  int res = first->item;
  first = first->next;
  itemsCount--;
  return res;
}

bool IntList::Remove(int value) {
  ListItem *prev = 0;
  ListItem *current = first;
  while (current) {
    if (current->item == value) {
      if (prev) {
        prev->next = current->next;
      }
      if (current == last) {
        last = prev;
      }
      delete current;
      itemsCount--;
      return true;
    } else {
      prev = current;
      current = current->next;
    }
  }
  return false;
}

void IntList::Insert(int value) {
  ListItem *prev = NULL;
  ListItem *succ = first;

  while (succ != NULL && succ->item < value) {
    prev = succ;
    succ = succ->next;
  }

  ListItem *newItem = new ListItem(value, succ);
  if (succ == NULL) {
    last = newItem;
  }
  if (prev == NULL) {
    first = newItem;
  } else {
    prev->next = newItem;
  }
  itemsCount++;
}

string IntList::GetAllItemsInfo() {
  stringstream stream;
  ListItem *current = first;
  while (current) {
    stream << current->item << ' ';
    current = current->next;
  }
  stream << endl;
  return stream.str();
}
