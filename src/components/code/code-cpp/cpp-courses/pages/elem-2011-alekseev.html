1 Таблица кодировки

Создадим таблицу кодировки с помощью ООП. Создадим класс CodeTable который хранит массив символов array. Поля lbound и hbound описывают нижнюю и верхнюю границу диапазона символов. В конструкторе задаются lbound и hbound и происходит инициализация таблицы преобразования массивом iniTable. Второй конструктор это конструктор копирования. Деструктор нужен для того чтобы освободить память от массива array. Определение оператора [] индексации.

#pragma once это защита от повторного включения заголовочного файла.

Файл codetable.h

#pragma once

typedef unsigned char byte;

class CodeTable {
  byte lBound;
  byte hBound;
  byte *array;

public:
  // конструктор
  CodeTable(byte low, byte high, byte* iniTable = 0);

  // конструктор копирования
  CodeTable(const CodeTable &src);

  // деструктор
  ~CodeTable();

  byte& operator[](byte i);
};

В конструкторе сначала проверяем что верхняя граница не ниже нижней границы. Потом выделяем память под массив array. В каждую ячейку массива записываем его индекс. Теперь таблица кодов будет преобразовываться код сам в себя. Далее проверяется условие if (iniTable). Если массив iniTable не задан, то условие if не выполняется и инициализация таблицы не выполняется. Иначе инициализируем массив array массивом iniTable.

В конструкторе копирования выделяется память под массив и в него копируются данные из переданного массива.

Деструктор освобождает память, которую коснтруктор динамически выделял под массив array с помощью оператора new.

Определение оператора [] - это операция индексации. Сначала проверяется что индекс попадает в границы между lbound и hbound. Потом из индекса вычитается значение нижней границы чтобы получить индекс элемента в массиве array.

#include <stdexcept> подключение стандартной библиотеки для поддержки типовых исключений.

Файл codetable.cpp

#include <stdexcept>
#include <string>
#include "codetable.h"

using namespace std;

// конструктор
CodeTable::CodeTable(byte low, byte high, byte* iniTable) {
  if ((hBound = high) < (lBound = low)) {
    throw out_of_range("CodeTable constructor: lower bound is higher than upper one");
  }
  array = new byte[high - low + 1];
  for (byte code = lBound; ; code++) {
    array[code - lBound] = code;
    if (code == hBound) break;
  }

  if (iniTable) {
    for (int ndx = 0; ndx <= high - lBound && iniTable[ndx]; ndx++) {
      array[ndx] = iniTable[ndx];
    }
  }
}

// конструктор копирования
CodeTable::CodeTable(const CodeTable &src) {
  array = new byte[(hBound = src.hBound) - (lBound = src.lBound) + 1];

  for (byte ndx = lBound; ; ndx++) {
    array[ndx - lBound] = src.array[ndx - lBound];
    if (ndx == hBound) break;
  }
}

// деструктор
CodeTable::~CodeTable() {
  delete[] array;
}

// операция индексации []
byte & CodeTable:: operator[] (byte i) {
  if (i < lBound || i > hBound) {
    throw out_of_range("Index is out of range");
  }
  return array[i - lBound];
}

2 Динамический массив

В статическом массиве число элементов изменять нельзя. Мы хотим создать динамический массив в котором можно увеличивать или уменьшать число элементов. Сделаем его на основе статического массива.

Пусть нам нужен динамический массив, который изначально содержит 5 элементов, тогда создадим статичкский массив из 10 элементов. Заведем переменную elementsCount = 5, которая указывает на то что в число элементов в массиве равно пяти, хотя память выделена под 10 элементов.

Добавим в этот массив еще один элемент, тогда elementsCount = 6.

Добавим еще 5 элементов и получится массив из 11 элементов. Но тогда для хранения 11 элементов нам не хватит размера массива из 10 элементов. Тогда создадим новый статический массив на 22 элемента. Скопируем в него 6 элементов из старого массива и добавим 5 новых элементов. Таким образом, в новом массиве будет elementsCount = 11. А старый массив удалим и освободим от него память.

template <class Elem> означает что класс DinamicArray является шаблонным. Это сделано для того чтобы в массиве можно было сохранять элементы любого типа данных. Поле size это количество элементов в массиве. Поле maxSize это максимальное количество элементов массива. Поле array это указатель на сам массив, который хранится в куче.

operator [] оператор индексации нужен чтобы к элементам DynamicArray мы могли обращаться как к обычному массиву через квадратные скобки.

Метод Resize изменяет размер массива.

Метод Add добавляет элемент в массив.

Метод Enlarge увеличивает размер массива.

Метод Shrink уменьшает размер массива.

В условии if ((size = sz) < 0) сначала переменной size присваивается значение переменной sz, а за тем проверяется что это значение меньше нуля. Если size меньше нуля то выбрасывается исключение.

Операция индексации проверяет что индекс не выходит за границы массива.

Метод Resize принимает параметр delta. Если delta больше нуля то вызывается метод Enlarge, если delta меньше нуля то вызывается метод Shrink.

Метод Add добавляет новый элемент в конец массива и увеличивает его размер на 1.

Метод Enlarge создает новый массив, большего размера. В условии if ((size += delta) > maxSize) сначала увеличивается переменная size на значение delta, а затем проверяется что это значение больше maxSize. Если больше, то выделяется место для нового массива и старый массив переносится в новое место.

Метод Shrink уменьшает число элементов size в массиве. Но не уменьшает место, выделенное под массив.

Файл DinamicArray.h

#pragma once

#include <stdexcept>

using namespace std;

template <class Elem>
class DynamicArray {
private:
  int size;
  int maxSize;
  Elem *array;

public:
  // конструктор
  DynamicArray(int sz = 0, int maxSz = 0, Elem *iniArray = NULL);

  // конструктор копирования
  DynamicArray(const DynamicArray<Elem> & a) {
    size = maxSize = 0;
    *this = a;
  }

  // деструктор
  ~DynamicArray() { delete[] array; }

  // оператор индексации
  Elem & operator [] (int i);

  // оператор присваивания
  DynamicArray<Elem> & operator = (const DynamicArray<Elem> & a);
  void Resize(int delta);
  void Add(const Elem & e);

private:
  void Enlarge(int delta);
  void Shrink(int delta);
};

// реализация конструктора класса
template <class Elem>
DynamicArray<Elem>::DynamicArray(int sz, int maxSz, Elem *iniArray) {
  if ((size = sz) < 0) {
    throw out_of_range("DynamicArray constructor : Negative array size");
  }
  maxSize = (maxSz < sz ? sz : maxSz);
  array = new Elem[maxSize];
  if (iniArray) {
    for (int i = 0; i < size; i++) {
      array[i] = iniArray[i];
    }
  }
}

// реализация операции индексации
template <class Elem>
Elem & DynamicArray<Elem>::operator [] (int i) {
  if (i < 0 || i >= size) {
    throw out_of_range("operator [] : Array index is out of range");
  }
  return array[i];
}

// реализация операции присваивания
template <class Elem>
DynamicArray<Elem> & DynamicArray<Elem>::operator=(const DynamicArray<Elem> & a) {
  Resize(a.size - size);
  for (int i = 0; i < size; i++) {
    array[i] = a.array[i];
  }
  return *this;
}

template <class Elem>
void DynamicArray<Elem>::Resize(int delta) {
  if (delta > 0) {
    Enlarge(delta);
  } else if (delta < 0) {
    Shrink(-delta);
  }
}

template <class Elem>
void DynamicArray<Elem>::Add(const Elem & e) {
  Resize(1);
  array[size-1] = e;
}

template <class Elem>
void DynamicArray<Elem>::Enlarge(int delta) {
  if ((size += delta) > maxSize) {
    maxSize = size;
    Elem *newArray = new Elem[maxSize];
    for (int i = 0; i < size - delta; i++) {
      newArray[i] = array[i];
    }
    array = newArray;
  }
}

template <class Elem>
void DynamicArray<Elem>::Shrink(int delta) {
  size = (delta > size ? 0: size- delta);
}

3 Связный список

Связный список это последовательность элементов, каждый из которых хранит значение элемента и указатель на следующий элемент. Последний элемент хранит указатель на NULL. Задается список всегда указателем на его первый элемент.

Двусвязный список это связный список, каждый элемент которого имеет еще один указатель на предыдущий элемент. Первый элемент вторым указателем указывает на NULL.

Чтобы создать список, нужно создать два типа данны. Первый тип данных для элемента списка. Его можно сделать шаблонным, чтобы список мог хранить данные любого типа. Второй тип данных для самого списка, которых хранит указатель на первый элемент списка и логику работы со списком, например, добавление и удаление элемента из списка.

Создадим класс IntList для создания списка из элементов типа int.

Внутри класса создадим структуру ListItem для создания элемента списка. Поле item хранит значение элемента. Указатель next указывает на следующий элемент списка. Конструктор задает начальные значения полям item и next при создании элемента.

Внутри класса имеем три поля. itemsCount это число элементов в списке. Это поле должно обновляться при добавлении или удалении элемента из списка. first это указатель на первый элемент списка. last это указатель на последний элемент списка.

Конструктор по умолчанию создает пустой связный список, где число элементов в списке 0 и указатели fist и last равны NULL.

Конструктор копирования создает полную копию связного списка, который приходит к нему как параметр.

Деструктор удаляет из памяти все элементы связного списка.

Метод Head возвращает значение первого элемента.

Метод Tail возвращает значение последнего элемента.

Метод AddLast принимающий в качестве параметра связный список, добавляет в конец списка другой связный список.

Метод AddFirst добавляет элемент в начало списка.

Метод AddLast принимающий в качестве параметра значение, добавляет элемент с этим значением в конец списка.

Метод RemoveFirst удаляет первый элемент и возвращает его значение.

Метод Remove удаляет элемент со значением n и возвращает true если элемент успешно удален.

Метод Insert вставляет элемент со значением n. Список получается упорядоченным.

Метод GetItemsCount возвращает число элементов.

Метод GetAllItemsInfo возвращает строку с информацией о элементах списка.

Файл IntList.h

#pragma once

#include <string>
#include <stdlib.h>

using namespace std;

class IntList {
private:
  struct ListItem {
    int item;
    ListItem *next;

    // конструктор
    ListItem(int i, ListItem *n = NULL) {
      item = i;
      next = n;
    }
  };

  int itemsCount;
  ListItem *first;
  ListItem *last;

public:
  // конструктор по умолчанию
  IntList() {
    itemsCount = 0;
    first = last = NULL;
  }

  // конструктор копирования
  IntList(const IntList & src);

  // деструктор
  ~IntList();

  int Head() const {
    return first->item;
  }

  int & Head() {
    return first->item;
  }

  int Tail() const {
    return last->item;
  }

  int & Tail() {
    return last->item;
  }

  void AddLast(const IntList & src);
  void AddFirst(int item);
  void AddLast(int item);
  int RemoveFirst();
  bool Remove(int n);
  void Insert(int n);

  int GetItemsCount() {
    return itemsCount;
  }

  string GetAllItemsInfo();
};

#include <sstream>
#include <string>
#include "IntList.h"

using namespace std;

// конструктор копирования
IntList::IntList(const IntList & src) {
  itemsCount = 0;
  first = last = NULL;
  AddLast(src);
}

// деструктор
IntList::~IntList() {
  ListItem *current = NULL;
  ListItem *next = first;
  while (next) {
    current = next;
    next = next->next;
    delete current;
  }
}

void IntList::AddLast(const IntList & src) {
  for (ListItem *cur = src.first; cur; cur = cur->next) {
    AddLast(cur->item);
  }
}

void IntList::AddFirst(int item) {
  ListItem *newItem = new ListItem(item, first);
  if (!first) {
    last = newItem;
  }
  first = newItem;
  itemsCount++;
}

void IntList::AddLast(int item) {
  ListItem *newItem = new ListItem(item);
  if (!last) {
    first = newItem;
  } else {
    last->next = newItem;
  }
  last = newItem;
  itemsCount++;
}

int IntList::RemoveFirst() {
  int res = first->item;
  first = first->next;
  itemsCount--;
  return res;
}

bool IntList::Remove(int value) {
  ListItem *prev = 0;
  ListItem *current = first;
  while (current) {
    if (current->item == value) {
      if (prev) {
        prev->next = current->next;
      }
      if (current == last) {
        last = prev;
      }
      delete current;
      itemsCount--;
      return true;
    } else {
      prev = current;
      current = current->next;
    }
  }
  return false;
}

void IntList::Insert(int value) {
  ListItem *prev = NULL;
  ListItem *succ = first;

  while (succ != NULL && succ->item < value) {
    prev = succ;
    succ = succ->next;
  }

  ListItem *newItem = new ListItem(value, succ);
  if (succ == NULL) {
    last = newItem;
  }
  if (prev == NULL) {
    first = newItem;
  } else {
    prev->next = newItem;
  }
  itemsCount++;
}

string IntList::GetAllItemsInfo() {
  stringstream stream;
  ListItem *current = first;
  while (current) {
    stream << current->item << ' ';
    current = current->next;
  }
  stream << endl;
  return stream.str();
}

Файл Tree.h

#pragma once

template <class T>
class Tree {
  struct Node {
    T item;
    Node *son;
    Node *brother;

    Node(T i, Node *s = NULL, Node *b = NULL) {
      item = i;
      son = s;
      brother = b;
    }
  };

  Node *root;

public:
  Tree() {
    root = NULL;
  }

  ~Tree() {
    DeleteSubtree(root);
  }

  int GetHeight() {
    return GetHeight(root);
  }

  int GetNodesOnLevel(int level) {
    return GetNodesOnLevel(root, level);
  }

  void Add(const T &elem) {
    Add(root, elem);
  }

private:
  void Add(Node *&node, const T &elem);
  void DeleteSubtree(Node *node);
  int GetHeight(Node *node);
  int GetNodesOnLevel(Node *node, int level);
};

template <class T>
void Tree<T>::DeleteSubtree(Node *node) {
  if (node) {
    DeleteSubtree(node->son);
    DeleteSubtree(node->brother);
    delete node;
  }
}

template <class T>
int Tree<T>::GetHeight(Node *node) {
  if (node == NULL) {
    return 0;
  }
  int max = 0;
  for (Node *current = node->son; current; current = current->brother) {
    int curHeight = GetHeight(current);
    if (curHeight > max) {
      max = curHeight;
    }
  }
  return max + 1;
}

template <class T>
int Tree<T>::GetNodesOnLevel(Node *node, int level) {
  if (node == NULL) {
    return 0;
  }
  if (level <= 0) {
    return 0;
  }
  return GetNodesOnLevel(node->son, level - 1) +
  (level == 1) +
  GetNodesOnLevel(node->brother, level);
}

template <class T>
void Tree<T>::Add(Node *&node, const T &item) {
  if (node == NULL) {
    node = new Node(item);
  } else if (item < node->item) {
    Add(node->son, item);
  } else {
    Add(node->brother, item);
  }
}

6 Пирамида (куча)

Пирамида (куча) это пронумерованное бинарное дерево. У каждого элемента может быть не более двух дочерних элементов (поддеревьев). Предок узла с номером i всегда является узел с номером i/2.

        1
     /    \
    2      3
   / \    / \
  4   5  6   7
 / \
8   9

Пирамиду будем хранить в массиве items. Массив статический, поэтому количество элементов в нем увеличить не получится.

Метод Insert добавляет элемент в кучу. При попытке вставить больше элементов, выдается исключение NoMoreSpace. При добавлении элемента он сначала становится в конец массива, затем всплывает на свое место.

#include "NoMoreSpace.h"

using namespace std;

template <class T>
class Heap {
private:
  int count;
  T *items;
  int size;

public:
  Heap(int size) {
    count = 0;
    items = new T[Heap::size = size];
  }

  void Insert(const T & element);

  T & operator [](int i) {
    return items[i];
  }

  const T & operator [](int i) const {
    return items[i];
  }

  void Print();
};

template <class T>
void Heap<T>::Insert(const T & elem) {
  if (count >= size) {
    throw NoMoreSpace();
  }
  int curIndex = count++;
  int parentIndex;

  while (curIndex > 0 && elem < items[parentIndex = (curIndex-1)/2]) {
    items[curIndex] = items[parentIndex];
    curIndex = parentIndex;
  }
  items[curIndex] = elem;
}

template <class T>
void Heap<T>::Print() {
  for (int i = 0; i < count; i++) {
    cout << items[i] << ' ';
  }
  cout << endl;
}

7 Множество

Множество это битовое поле, в котором для доступа к элементу используется не его имя, а индекс. Похоже на массив, но каждый элемент всегда занимает один бит. Основные операции:

  • Проверка принадлежности элемента множеству
  • Объединение множеств
  • Пересечение множеств
  • Добавление элемента в множество
  • Удаление элемента из множества
  • Определение числа элементов множества (мощность множества)

#include <stdexcept>
#include "Set.h"

using namespace std;

Set::Set(int min, int max) {
  if (min > max) {
    minElem = max;
    maxElem = min;
  } else {
    minElem = min;
    maxElem = max;
  }
  int num = maxElem - minElem + 1;
  numWords = (num + 15) >> 4;
  elems = new WORD[numWords];

  for (int i = 0; i < numWords; i++) {
    elems[i] = 0;
  }
}

Set::Set(const Set & s) {
  minElem = s.minElem;
  maxElem = s.maxElem;
  elems = new WORD[numWords = s.numWords];

  for (int i = 0; i < numWords; i++) {
    elems[i] = s.elems[i];
  }
}

Set::~Set() {
  delete [] elems;
}

bool Set::Has(int n) const {
  if (n > maxElem || n < minElem) {
    return false;
  }

  int word = (n - minElem) >> 4;
  int bit = (n - minElem) & 15;
  return (elems[word] & (1 << bit)) != 0;
}

// добавление элемента в множество
Set & Set::operator |= (int n) {
  if (n <= maxElem && n >= minElem) {
    int word = (n - minElem) >> 4;
    int bit = (n - minElem) & 15;
    elems[word] |= (1 << bit); // записываем в нужный бит 1
  } else {
    throw out_of_range("Cannot add an element: it is out of range");
  }
  return *this;
}

// добавление множества в множество
Set & Set:: operator |= (const Set & other) {
  if (other.minElem != minElem || other.maxElem != maxElem) {
    throw out_of_range("Set incomparable");
  }
  for (int i = 0; i < numWords; i++) {
    elems[i] |= other.elems[i];
  }
  return *this;
}

// пересечение множеств
Set & Set::operator &= (const Set & other) {
  if (other.minElem != minElem || other.maxElem != maxElem) {
    throw out_of_range("Set incomparable");
  }
  for (int i = 0; i < numWords; i++) {
    elems[i] &= other.elems[i];
  }
  return *this;
}

// удаление элемента из множества
Set & Set::operator -= (int n) {
  if (n <= maxElem && n >= minElem) {
    int word = (n - minElem) >> 4;
    int bit = (n - minElem) & 15;
    elems[word] &= ~(1 << bit); // записываем в нужный бит 0
  }
  return *this;
}

// удаление множества из множества
Set & Set:: operator -= (const Set & other) {
  if (other.minElem != minElem || other.maxElem != maxElem) {
    throw out_of_range("Sets incomparable");
  }
  for (int i = 0; i < numWords; i++) {
    elems[i] &= ~other.elems[i];
  }
  return *this;
}

// инверсия элементов
Set & Set::Inverse() {
  for (int i = 0; i < numWords; i++) {
    elems[i] = ~elems[i];
  }
  return *this;
}

}

}

Массив monthLength хранит число дней, прошедших с начала года до начала месяца.

Функция Convert конвертирует дату в виде дня, месяца, года в число дней прошедших с 1 января 1600 года. Функция Convert имеет перегруженный вариант, который конвертирует число дней прошедших с 1 января 1600 года в дату в виде дня, месяца, года.

Функция CorrectLastMonthDate следит чтобы при изменении месяца, день не мог стать больше числа дней в месяце.

static long Convert(int d, int m, int y) {
  int year = y - 1600;
  long days = year*365 + year/4 - year/100 + year/400;
  bool leap = IsLeapYear(y);
  if (leap) {
    days--;
  }

  return days + monthLength[leap][m-1] + d;
}

static void Convert(long days, int & d, int & m, int & y) {
  y = (int)(1600 + 4*days/(365*4 + 1)) + 1;
  int diff = (int)(days - Convert(1, JANUARY, y) + 1);
  if (diff <= 0) {
    y--;
  }
  bool leap = IsLeapYear(y);
  if (diff <= 0) {
    diff += 365 + leap;

  for (m = JANUARY;
  m < DECEMBER && diff > monthLength[leap][m];
  m++) ;

  d = diff - monthLength[leap][m-1];
}

#pragma once

template <class Key>
int BinSearch(Key * array, int low, int high, const Key & key) {
  while (low < high) {
    int middle = (low + high) / 2;
    if (array[middle] >= key) {
      high = middle;
    } else {
      low = middle + 1;
    }
  }
  return array[low] == key ? low : -1;
}

template <class Key>
void Insert(Key * array, int low, int high, const Key & key) {
  if (array[high] <= key) {
    array[high+1] = key;
  } else {
    int border = high;

    while (low < high) {
      int middle = (low + high) / 2;
      if (array[middle] >= key) {
        high = middle;
      } else {
        low = middle + 1;
      }
    }

    for (int i = border; i >= low; i--) {
      array[i+1] = array[i];
    }

    array[low] = key;
  }
}

void Merge(Key * arr1, int low1, int high1,
           Key * arr2, int low2, int high2,
           Key * arr3, int low3) {
  int ndx1 = low1;
  int ndx2 = low2;
  int ndx3 = low3;

  while (ndx <= high1 && ndx2 <= high2) {
    if (arr1[ndx1] < arr2[ndx2]) {
      arr3[ndx3++] = arr1[ndx1++];
    } else {
      arr3[ndx3++] = arr2[ndx2++];
    }
  }

  if (ndx1 > high1) {
    while (ndx2 <= high2) {
      arr3[ndx3++] = arr2[ndx2++];
    } else {
      while (ndx1 <= high1) {
        arr3[ndx3++] = arr1[ndx1++];
      }
    }
  }
}

#pragma once

template <class Key>
void HeapSort(Key * array, int low, int high) {
  int heapLength = 1;

  // пока весь массив не станет частью пирамиды
  // заполняем пирамиду элементами массива
  while (heapLength <= high-low) {
    int ndx = low + heapLength;
    Key elem = array[ndx];
    while (ndx > low && elem > array[(ndx - low - 1)/2]) {
      array[ndx] = array[(ndx - low - 1)/2];
      ndx = (ndx - low - 1)/2;
    }
    array[ndx] = elem;
    heapLength++;
  }

  // пока в пирамиде остаются элементы
  // перетаскиваем их в массив
  while (heapLength > 1) {
    heapLength--;
    Key elem = array[low + heapLength];
    array[low + heapLength] = array[low];
    int ndx = low;
    while (2*(ndx-low) < heapLength-1) {
      int child1 = low + 2*(ndx - low) + 1;
      int child2 = low + 2*(ndx - low) + 2;
      if (child2 <= low + heapLength - 1 &&
          array[child2] > array[child1]) {
            child1 = child2;
      }
      if (elem > array[child1]) {
        break;
      }
      array[ndx] = array[child1];
      ndx = child1;
    }
    array[ndx] = elem;
  }
}

#pragma once

#include "InsertSort.h"

template <class Key>
void QuickSort(Key * array, int low, int high) {
  int pLow = low;
  int pHigh = high;
  Key elem = array[low];

  while (pLow != pHigh) {
    while (pHigh > pLow && array[pHigh] >= elem) {
      pHigh--;
    }

    if (pHigh > pLow) {
      array[pLow] = array[pHigh];

      while (pLow < pHigh && array[pLow] <= elem) {
        pLow++;
      }
      array[pHigh] = array[pLow];
    }
  }

  array[pLow] = elem;
  if (pLow - low > 1) {
    if (pLow - low > 10) {
      QuickSort<Key>(array, low, pLow-1);
    } else {
      InsertSort<Key>(array, low, pLow-1);
    }
  }

  if (high - pHigh > 1) {
    if (high - pHigh > 10) {
      QuickSort<Key>(array, pHigh+1, high);
    } else {
      InsertSort<Key>(array, pHigh+1, high);
    }
  }
}
