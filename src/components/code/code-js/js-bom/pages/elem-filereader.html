Пример

Создадим в index.html input для выбора файла и div для вывода текста из файла.

<input type="file" id="myFile"> <input type="file" id="myFiles" multiple> <div id="content"></div>

В index.js запишем код

const output = document.getElementById('content');

// обработаем выбор одного файла
document.getElementById('myFile').onchange = getFile;
function getFile(e) {
  const file = e.target.files[0];
  console.log(file.name, file.type, file.size);
  const reader = new FileReader();
  reader.onload = function(ev) {
    output.textContent = ev.target.result;
  };
  reader.readAsText(file);
}

// обработаем выбор нескольких файлов
document.getElementById('myFiles').onchange = getFiles;
function getFiles(e) {
  const files = e.target.files;
  for (let i = 0; i < files.length; i++) {
    const file = files[i];
    console.log(file.name, file.type, file.size);
    const reader = new FileReader();
    reader.onload = function(ev) {
      output.textContent += ev.target.result;
    };
    reader.readAsText(file);
  }
}
