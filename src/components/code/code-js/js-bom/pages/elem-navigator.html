navigator.geolocation

  1. getCurrentPosition()

    getCurrentPosition() - это асинхронный метод, который принимает три параметра.

    Первый - callback функция, которая запустится после того как браузер получит местоположение. Она принимает один параметр - объект, содержащий два свойства timestamp и coords. coords это объект со свойствами latitude - широта, longitude - долгота, accuracy - точность.

    Второй - callback функция, которая запустится если будет ошибка. Она принимает два параметра - message сообщение и code код ошибки. Код ошибки: 1 - нет разрешения, 2 - техническая ошибка, 3- превышено время ожидания.

    Третий это объект с дополнительными опциями enableHighAccuracy, timeout, maximumAge.

    function success(position) {
  console.log(
    position.timestamp,
    position.coords.latitude,
    position.coords.longitude,
    position.coords.accuracy
  )
}

function error(err) {
  let s = '';
  switch(err) {
    case 1: s = 'Нет разрешения'; break;
    case 2: s = 'Техническая ошибка'; break;
    case 3: s = 'Превышено время ожидания'; break;
    default: s = 'Неизвестная ошибка'
  }
  console.log(s);
}

let params = {
  enableHighAccuracy = true,
  timeout = 10000,
  maximumAge = 0
}

navigator.geolocation.getCurrentPosition(success, error, params)
