Приоритет операций

Приоритет операторов

Приоритет операторов определяет порядок, в котором выполняются операции. Операции с более высоким приоритетом выполняются первыми.

Ассоциативность

Ассоциативность определяет порядок, в котором операции с одинаковым приоритетом обрабатываются. Левая ассоциативность (left-to-right) означает порядок слева направо. Правая ассоциативность (right-to-left) означает порядок справа налево.

Приоритет операций

  1. ( ... ) Группировка

    (5 + 3) * 2 // 16

  2. ... . ... Доступ к свойствам

    let a = {b: 5}; a.b // 5

    ... [ ... ] Доступ к свойствам с возможностью вычисления

    let a = {b: 5}; a['b'] // 5

    new ... ( ... ) new со списком аргументов

    let a = new Array(1,2,3); a // [1,2,3]

    ... ( ... ) Вызов функции

    Math.max(5,4) // 5

    ?. Оператор опциональной последовательности

  3. new ... new без списка аргументов (справа налево)

    let a = new Array; a // []

  4. ... ++ Постфиксный инкремент

    let i = 5; i++ // 6

    ... -- Постфиксный декремент

    let i = 5; i-- // 4
  5. (справа налево)

    ! ... Логическое отрицание

    !true // false !0 // true

    ~ ... Побитовое отрицание (инвертирование битов)

    let i = 5; ~i // -6 let j = -6; ~j // 5 let n = 10; ~n // -11 let k = -11; ~k // 10

    + ... Унарный плюс (преобразование в число)

    let i = '5'; +i // 5 let i = true; +i // 1

    - ... Унарный минус (отрицание числа)

    let i = 5; -i // -5 let i = -5; -i // 5

    ++ ... Префиксный инкремент

    let i = 5; ++i // 6

    -- ... Префиксный декремент

    let i = 5; --i // 4

    typeof ... Тип операнда

    typeof 5 // 'number' typeof 'a' // 'string' typeof true // 'boolean'

    void ... Возвращение значения undefined

    void true // undefined void 5 // undefined

    delete ... Удаление свойства

    let a = {b: 1, c: 2}; delete a.b; a // {c: 2}

    await ... Ожидание

  6. ... ** ... Возведение в степень (справа налево)

    5 ** 2 // 25 25 ** 2 // 625

  7. ... * ... Умножение

    5 * 2 // 10

    ... / ... Деление

    100 / 2 // 50

    ... % ... Остаток от деления

    5 % 2 // 1

  8. ... + ... Сложение

    5 + 5 // 10

    ... - ... Вычитание

    10 - 2 // 8

    ... + ... Конкатенация строк

    'hello' + ' world' // 'hello world'

  9. << Побитовый сдвиг влево

    1 << 1 // 2 3 << 1 // 6

    >> Побитовый сдвиг вправо с расширением знака

    2 >> 1 // 1 3 >> 1 // 1 4 >> 1 // 2 -4 >> 1 // -2

    >>> Побитовый сдвиг вправо с дополнением нулями

    2 >>> 1 // 1 3 >>> 1 // 1 4 >>> 1 // 2 -4 >> 1 // 2147483646

  10. ... < ... Меньше

    4 < 5 // true

    ... <= ... Меньше или равно

    4 <= 5 // true

    ... > ... Больше

    5 > 4 // true

    ... >= ... Больше или равно

    5 >= 4 // true

    ... in ... Проверка существует ли свойство

    let a = {b: 1}; 'b' in a // true

    ... instanceof ... Проверка класса объекта

    let a = {}; a instanceof Object // true

  11. ... == ... Проверка на нестрогое равенство

    1 == 1 // true '1' == 1 // true

    ... != ... Проверка на нестрогое неравенство

    1 != 1 // false '1' != 1 // false

    ... === ... Проверка на строгое равенство

    1 == 1 // true '1' == 1 // false

    ... !== ... Проверка на строгое неравенство

    1 !== 1 // false '1' !== 1 // true

  12. ... & ... Побитовое И

    7 & 7 // 7 7 & 5 // 5 7 & 0 // 0

  13. ... ^ ... Побитовое исключающее ИЛИ

    7 ^ 7 // 0 7 ^ 0 // 7

  14. ... | ... Побитовое ИЛИ

    7 | 7 // 7 7 | 5 // 7 7 | 0 // 7

  15. ... && ... Логическое И

    1 && 5 && 10 // 10 0 && 5 && 10 // 0 1 && false && 10 // false

  16. ... || ... Логическое ИЛИ

    1 || 5 || 10 // 1 0 || 5 || 10 // 5 0 || false || 10 // 10 undefined || Nan || null // null

    ... ?? ... Оператор нулевого слияния (выбор первого определенного операнда)

    5 ?? 0 // 5 0 ?? 5 // 0 null ?? 5 // 5 undefined ?? 5 // 5

  17. ... = ... Присваивание переменной или свойства

    let a = 5; a // 5 let a = {}; a.b = 5; a.b // 5

    ... += ..., ... -= ..., ... **= ..., ... *= ..., ... /= ..., ... %= ..., ... <<= ..., ... >>=... , ... >>>= ..., ... &= ..., ... ^= ..., ... |=..., ... &&= ..., ...||= ..., ... ??= ... Оперирование и присваивание

    let a = 10; a += 10 // 20 let a = 10; a -= 2 // 8

    ... ? ... : ... Тернарный оператор

    let a = 5 > 4 ? 5 : 4; a // 5 let a = 5 < 4 ? 5 : 4; a // 4

    yield ...

    yield* ...

  18. ... , ... Запятая (возвращение последнего операнда), последовательность

    5, 4 // 4 5, 4, 0 // 0 5, 4, 0, null, undefined // undefined

    В цепочке вычисляет каждый операнд, но возвращает только последний
