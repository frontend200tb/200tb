Свойство __proto__ есть у всех объектов

И __proto__ и prototype это свойства объектов. Выясним у каких объектов есть свойство __proto__, а у каких prototype.

  1. man это объект, значит у него есть __proto__ let man = {} // man.__proto__
  2. users это массив, а массив это объект, значит у него есть __proto__ let users = [] // users.__proto__
  3. age это примитив, это не объект. Но если мы к примитиву ставим точку, то в памяти временно примитив оборачивается в объект и это становится объект, значит у него есть __proto__ let age = 18 // age.__proto__
  4. Аналогично строка. let str = 'qwert' // str.__proto__
  5. Аналогично булевое значение. let isMan = true // isMan.__proto__
  6. Когда мы создаем функцию с ключевым словом function при помощи function declaration, создается объект function run() {} // run.__proto__
  7. Когда мы создаем анонимную функцию с ключевым словом function при помощи function expression, создается переменная, которая ссылается на объект const eat = function() {} // eat.__proto__
  8. Когда мы создаем стрелочную функцию, создается переменная, которая ссылается на объект const jump = () => {} // jump.__proto__
  9. Класс это функция, а функция это объект class person {} // person.__proto__

__proto__ это объект

Свойство __proto__ у однотипных объектов ссылаются на один и тот же объект

  1. Объекты, созданные при помощи литерала объекта 
    const man1 = {};
const man2 = {};
console.log(man1.__proto__ === man2.__proto__) // true
  2. Массивы, созданные при помощи литерала массива 
    const users = [];
const cars = [];
console.log(users.__proto__ === users.__proto__) // true
  3. Числа, созданные при помощи литерала числа 
    const num1 = 18;
const num2 = 100;
console.log(num1.__proto__ === num2.__proto__) // true
  4. Строки, созданные при помощи литерала строки 
    const str1 = 'asdf';
const str2 = 'qwer';
console.log(str1.__proto__ === str2.__proto__) // true
  5. Булевые переменные, созданные при помощи литерала 
    const bul1 = true;
const bul2 = false;
console.log(bul1.__proto__ === bul2.__proto__) // true
  6. Все функции и классы и меют один __proto__ 
    function run() {};
const eat = function() {};
const jump = () => {};
class person {};
console.log(run.__proto__ === eat.__proto__) // true
console.log(jump.__proto__ === person.__proto__) // true

Любой объект создается при помощи функции-конструктора

Любой объект создается при помощи функции-конструктора и ключевого слова new или при помощи класса. Если мы этого не видим, то все равно это происходит под капотом. Если мы создаем объект явно при помощи слова new, то в свойство __proto__ созданного объекта попадет функция-конструктор (тоже объект), с помощью которой он создан.

Когда мы создаем примитив, то он не создается при помощи ключевого слова new. Но когда мы к нему обращаемся через точку, то временно происходит создание объекта при помощи слова new

  1. let promise = new Promise(() => {}) // new Promise(...)
  2. let man = {} // new Object(...)
  3. let users = [] // new Array(...)
  4. let num = 18 // new Number(...)
  5. let str = 'asdf' // new String(...)
  6. let bul = true // new Boolean(...)
  7. function run() {} // new Function(...) const eat = function() {} // new Function(...) const jump = () => {} // new Function(...) class person {} // new Function(...)

Любой класс и любая функция имеет prototype

Любой класс и функция, созданная с ключевым слово function имеет прототип. Прототипы разных классов и функций это разные объекты со своими свойствами и методами и они не равны друг другу.

class Person {}
function Run() {}
const Eat = function() {}

__proto__ vs prototype

  1. __proto__ есть у любого объекта
  2. любой объект создается с помощью class или function
  3. prototype есть только у class или function
  4. __proto__ любого объекта ссылается на prototype класса (или функции-конструктора), с помощью которого он был создан
  5. __proto__ и prototype это свойства объекта
  1. let promise = new Promise(() => {}) // new Promise(...) promise.__proto__ === Promise.prototype
  2. let man = {} // new Object(...) man.__proto__ === Object.prototype
  3. let users = [] // new Array(...) users.__proto__ === Array.prototype
  4. let num = 18 // new Number(...) num.__proto__ === Number.prototype
  5. let str = 'asdf' // new String(...) str.__proto__ === String.prototype
  6. let bul = true // new Boolean(...) bul.__proto__ === Boolean.prototype
  7. function run() {} // new Function(...) run.__proto__ === Function.prototype const eat = function() {} // new Function(...) eat.__proto__ === Function.prototype const jump = () => {} // new Function(...) jump.__proto__ === Function.prototype class person {} // new Function(...) person.__proto__ === Function.prototype
