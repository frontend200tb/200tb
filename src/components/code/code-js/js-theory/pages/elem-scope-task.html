Задача 4. Что будет в консоли?

var o = {};
var i = 0;

(function () {
  o.a = 'a';
  i++;
})();
console.log(o, i);

(function (o,i) {
  o.b = 'b';
  i++;
})(o, i);
console.log(o, i);
Ответ { a: 'a' } 1 { a: 'a', b: 'b' } 1
