Задача 1

Треугольник задан на плоскости координатами своих вершин. Найти его периметр

#include <iostream>

int main()
{
  double x1, x2, x3, y1, y2, y3, a, b, c, p;

  std::cout << "Введите первую вершину" << std::endl;
  std::cin >> x1 >> y1;
  std::cout << "Введите вторую вершину" << std::endl;
  std::cin >> x2 >> y2;
  std::cout << "Введите третью вершину" << std::endl;
  std::cin >> x3 >> y3;

  a = sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1));
  b = sqrt((x2-x3)*(x2-x3) + (y2-y3)*(y2-y3));
  c = sqrt((x1-x3)*(x1-x3) + (y1-y3)*(y1-y3));
  p = a+b+c;

  std::cout << "Периметр " << p << std::endl;
  return 0;
}
