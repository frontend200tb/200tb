Объявление и инициализация двумерного массива

int* nums = new int[3]{1,2,3}; 
#include <iostream>

int main()
{
  const int n = 3; // количество строк
  const int m = 4; // количество столбцов

  int **nums = new int*[m]{};
  std::cout << nums[1] << std::endl;
  for (int i = 0; i < n; i++)
  {
    nums[i] = new int[m]();
  }
  std::cout << nums[1] << std::endl;

  delete[] nums;
}
