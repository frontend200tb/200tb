Question 10/12

Implement a function sum(num1)(num2)...() which can be called endlessly. Example

console.log(sum(2)()); // 2 console.log(sum(2)(3)()); // 5 console.log(sum(2)(3)(4)()); // 9 
function sum(num) {
  let summa = num;

  function result(next) {
    if (next === undefined) {
      return summa;
    }
    summa += next;
    return result;
  }

  return result;
}
